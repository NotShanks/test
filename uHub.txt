local REAL_SECONDS_PER_DAY = 600
local IN_GAME_MINUTES_PER_DAY = 1440
local REAL_SECONDS_PER_GAME_MIN = REAL_SECONDS_PER_DAY / IN_GAME_MINUTES_PER_DAY 

local function getSeasonTimerText()
	local effectsFolder = workspace:FindFirstChild("Effects")
	local dayValueObj = effectsFolder and effectsFolder:FindFirstChild("Day")
	local ui = Player:FindFirstChild("PlayerGui")
	local timeTextObj = ui and ui:FindFirstChild("Moves") and ui.Moves:FindFirstChild("Bars") and ui.Moves.Bars:FindFirstChild("Time")

	if not dayValueObj or not timeTextObj then
		return "Error loading data"
	end

	local currentDay = dayValueObj.Value
	local remainder = currentDay % 8
	if remainder == 0 then remainder = 8 end

	local actionText = ""
	local daysDifference = 0

	if remainder == 3 or remainder == 4 then
		actionText = "Fall ends"; daysDifference = 5 - remainder
	elseif remainder == 7 or remainder == 8 then
		actionText = "Spring ends"; daysDifference = 9 - remainder
	elseif remainder == 1 or remainder == 2 then
		actionText = "Fall starts"; daysDifference = 3 - remainder 
	elseif remainder == 5 or remainder == 6 then
		actionText = "Spring starts"; daysDifference = 7 - remainder 
	end

	local timeText = timeTextObj.Text
	local hoursStr, minutesStr = string.match(timeText, "(%d+):(%d+)")
	
	if not hoursStr or not minutesStr then return "Time parsing error" end

	local currentTotalGameMins = (tonumber(hoursStr) * 60) + tonumber(minutesStr)
	local gameMinsRemainingToday = IN_GAME_MINUTES_PER_DAY - currentTotalGameMins
	local totalGameMinsLeft = gameMinsRemainingToday + ((daysDifference - 1) * IN_GAME_MINUTES_PER_DAY)
	local realSecondsLeft = totalGameMinsLeft * REAL_SECONDS_PER_GAME_MIN

	local minutes = math.floor(realSecondsLeft / 60)
	local seconds = math.floor(realSecondsLeft % 60)
	return string.format("%s in %dm %ds", actionText, minutes, seconds)
end

local function updateCurrentServerCache()
	if isInDungeon() or game.JobId == "" then return end
	local effects = workspace:FindFirstChild("Effects")
	local dayObj = effects and effects:FindFirstChild("Day")
	local ui = Player:FindFirstChild("PlayerGui")

	if dayObj and timeTextObj then
		local currentDay = dayObj.Value
		local timeText = timeTextObj.Text
		local hoursStr, minutesStr = string.match(timeText, "(%d+):(%d+)")
		if hoursStr and minutesStr then
			local gameMins = (tonumber(hoursStr) * 60) + tonumber(minutesStr)
			
			local cache = {}
			ensureConfigFolder()
			if isfile and isfile(cacheFileName) then
				pcall(function()
					cache = HttpService:JSONDecode(readfile(cacheFileName))
				end)
			end
			
			cache[game.JobId] = {
				day = currentDay,
				gameMins = gameMins,
				realTime = os.time()
			}
			
			-- Evict cache listings older than 12 hours
			local now = os.time()
			for job, data in pairs(cache) do
				if now - data.realTime > 43200 then
					cache[job] = nil
				end
			end
			
			pcall(function()
				ensureConfigFolder()
				writefile(cacheFileName, HttpService:JSONEncode(cache))
			end)
		end
	end
end

local function getTargetServerFromCache()
	if not isfile or not isfile(cacheFileName) then return nil end
	
	local cache = {}
	pcall(function()
		cache = HttpService:JSONDecode(readfile(cacheFileName))
	end)
	
	local now = os.time()
	local bestJobId = nil
	local bestTimeRemaining = 9999
	
	for jobId, data in pairs(cache) do
		if jobId ~= game.JobId then
			local elapsedRealSeconds = now - data.realTime
			local elapsedGameMins = (elapsedRealSeconds / REAL_SECONDS_PER_DAY) * IN_GAME_MINUTES_PER_DAY
			local projectedTotalMins = data.gameMins + elapsedGameMins
			
			local totalGameDaysElapsed = math.floor(projectedTotalMins / IN_GAME_MINUTES_PER_DAY)
			local finalProjectedDay = data.day + totalGameDaysElapsed
			local currentMinsInDay = projectedTotalMins % IN_GAME_MINUTES_PER_DAY
			
			local remainder = finalProjectedDay % 8
			if remainder == 0 then remainder = 8 end
			
			local isFallOrSpring = (remainder == 3 or remainder == 4 or remainder == 7 or remainder == 8)
			
			local daysDifference = 0
			if remainder == 1 or remainder == 2 then
				daysDifference = 3 - remainder
			elseif remainder == 5 or remainder == 6 then
				daysDifference = 7 - remainder
			elseif remainder == 3 or remainder == 4 then
				daysDifference = 5 - remainder
			elseif remainder == 7 or remainder == 8 then
				daysDifference = 9 - remainder
			end
			
			local gameMinsRemainingToday = IN_GAME_MINUTES_PER_DAY - currentMinsInDay
			local totalGameMinsLeft = gameMinsRemainingToday + ((daysDifference - 1) * IN_GAME_MINUTES_PER_DAY)
			local realSecondsLeft = totalGameMinsLeft * REAL_SECONDS_PER_GAME_MIN
			
			if isFallOrSpring or realSecondsLeft <= 120 then
				if realSecondsLeft < bestTimeRemaining then
					bestTimeRemaining = realSecondsLeft
					bestJobId = jobId
				end
			end
		end
	end
	
	return bestJobId
end

local function doServerHop()
	local TeleportService = game:GetService("TeleportService")
	local PlaceId = game.PlaceId
	local JobId = game.JobId
	
	local success, result = pcall(function()
		local servers = {}
		-- Fetches public servers while excluding full instances
		local req = game:HttpGet("https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Desc&limit=100&excludeFullGames=true")
		local body = HttpService:JSONDecode(req)

		if body and body.data then
			for _, v in pairs(body.data) do
				if type(v) == "table" and tonumber(v.playing) and tonumber(v.maxPlayers) then
					if v.playing < v.maxPlayers and v.id ~= JobId then
						table.insert(servers, v.id)
					end
				end
			end
		end

		if #servers > 0 then
			local selectedInstance = servers[math.random(1, #servers)]
			TeleportService:TeleportToPlaceInstance(PlaceId, selectedInstance, Player)
		else
			-- Fallback to re-teleporting directly via PlaceId if server lookup fails
			TeleportService:Teleport(PlaceId, Player)
		end
	end)
	
	if not success then
		warn("Utility Hub Hopper Error: " .. tostring(result))
		pcall(function()
			TeleportService:Teleport(PlaceId, Player)
		end)
	end
end

local function executeProjectedServerHop()
	local cachedJobId = getTargetServerFromCache()
	if cachedJobId then
		updateUIRunsStatus("Hopping to cached target server...")
		local TeleportService = game:GetService("TeleportService")
		pcall(function()
			TeleportService:TeleportToPlaceInstance(game.PlaceId, cachedJobId, Player)
		end)
	else
		doServerHop()
	end
end

	local function tpWalkTo(targetVec)
		local reached = false
		local tpConnection

		tpConnection = RunService.Heartbeat:Connect(function(delta)
			if _G.CancelTween or not char or not hrp or not hrp.Parent then
				reached = true
				if tpConnection then tpConnection:Disconnect() end
				return
			end

			local currentPos = hrp.Position
			local dist = (targetVec - currentPos).Magnitude

			if dist < 5 then
				reached = true
				if tpConnection then tpConnection:Disconnect() end
				return
			end

			local direction = (targetVec - currentPos).Unit
			local moveAmount = direction * tpSpeed * delta * 10

			if moveAmount.Magnitude > dist then
				char:TranslateBy(direction * dist)
				reached = true
				if tpConnection then tpConnection:Disconnect() end
			else
				char:TranslateBy(moveAmount)
			end
			
			hrp.Velocity = Vector3.zero
			hrp.AssemblyLinearVelocity = Vector3.zero
		end)

		while not reached do
			task.wait()
		end
	end

	if not _G.CancelTween then
		tpWalkTo(hrp.Position + Vector3.new(0, 300, 0))
	end
	if not _G.CancelTween then
		tpWalkTo(Vector3.new(targetPos.X, hrp.Position.Y + 300, targetPos.Z)) 
	end
	if not _G.CancelTween then
		tpWalkTo(targetPos + Vector3.new(0, 3, 0))
	end

	if bv then bv:Destroy() end
	if noclipConn then noclipConn:Disconnect() end
	if humanoid then humanoid.PlatformStand = false end
	
	if hrp then
		hrp.Velocity = Vector3.zero
		hrp.AssemblyLinearVelocity = Vector3.zero
		hrp.AssemblyAngularVelocity = Vector3.zero
	end
end

local activeNoclip = nil
local activeFloat = nil
local currentTweenObj = nil

local function CancelSmoothTween()
	if currentTweenObj then currentTweenObj:Cancel() currentTweenObj = nil end
	if activeNoclip then activeNoclip:Disconnect() activeNoclip = nil end
	if activeFloat then activeFloat:Destroy() activeFloat = nil end
	
	local char = Player.Character
	if char and char:FindFirstChild("HumanoidRootPart") then
		stopMomentum(char.HumanoidRootPart)
	end
end

local function SmoothTween(targetCFrame, speed, conditionCheck)
	local char = Player.Character
	if not char then return false end
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not hrp then return false end

	CancelSmoothTween()

	local dist = (hrp.Position - targetCFrame.Position).Magnitude
	local tweenTime = dist / speed
	
	activeFloat = Instance.new("BodyVelocity")
	activeFloat.Name = "SmoothFloat"
	activeFloat.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
	activeFloat.Velocity = Vector3.zero
	activeFloat.Parent = hrp

	activeNoclip = RunService.Stepped:Connect(function()
		for _, v in pairs(char:GetDescendants()) do
			if v:IsA("BasePart") then v.CanCollide = false end
		end
	end)

	local tweenInfo = TweenInfo.new(tweenTime, Enum.EasingStyle.Linear)
	currentTweenObj = TweenService:Create(hrp, tweenInfo, {CFrame = targetCFrame})
	currentTweenObj:Play()
	
	local startTime = tick()
	while tick() - startTime < tweenTime do
		if conditionCheck and not conditionCheck() then
			CancelSmoothTween()
			return false
		end
		task.wait(0.05)
	end
	
	CancelSmoothTween()
	return true
end

local function getGates()
	local list = {}
	local refs = {}
	local gatesFolder = workspace:FindFirstChild("Gates")
	if gatesFolder then
		for _, gateModel in pairs(gatesFolder:GetChildren()) do
			if gateModel:IsA("Model") then
				local levelObj = gateModel:FindFirstChild("Level", true)
				local rankObj = gateModel:FindFirstChild("Rank", true)
				
				local timeLeftStr = ""
				local timeSpawnedObj = gateModel:FindFirstChild("TimeSpawned", true)
				
				if timeSpawnedObj then
					local spawnTime = tonumber(timeSpawnedObj.Value)
					if spawnTime then
						local remaining = 550 - (os.time() - spawnTime)
						if remaining > 0 then
							local mins = math.floor(remaining / 60)
							local secs = remaining % 60
							timeLeftStr = string.format(" | %02d:%02d", mins, secs)
						else
							timeLeftStr = " | Expired"
						end
					end
				end

				local lvl = levelObj and tostring(levelObj.Value) or "?"
				local rnk = rankObj and tostring(rankObj.Value) or "?"
				local name = string.format("%s [Lvl %s | Rank %s]%s", gateModel.Name, lvl, rnk, timeLeftStr)
				
				table.insert(list, name)
				refs[name] = gateModel
			end
		end
	end
	if #list == 0 then table.insert(list, "No Gates Found") end
	return list, refs
end

		local iconData = not isStudio and game:HttpGet("https://raw.githubusercontent.com/latte-soft/lucide-roblox/refs/heads/master/lib/Icons.luau")

		if not isStudio then
			icon = string.match(string.lower(icon), "^%s*(.*)%s*$")
			local sizedicons = icons['48px']
			local r = sizedicons[icon]
			if not r then return "rbxassetid://10723434557" end
			local irs = Vector2.new(r[2][1], r[2][2])
			local iro = Vector2.new(r[3][1], r[3][2])
			return { id = r[1], imageRectSize = irs, imageRectOffset = iro }

			return "rbxassetid://10723434557"
		end

		else

		end
	end

local function BlurModule(Frame)
	local RunService = game:GetService('RunService')
	local camera = workspace.CurrentCamera
	local MTREL = "Glass"
	local binds = {}
	local root = Instance.new('Folder', camera)
	root.Name = 'LunaBlur'

	local gTokenMH = 99999999
	local gToken = math.random(1, gTokenMH)

	local DepthOfField = Instance.new('DepthOfFieldEffect', game:GetService('Lighting'))
	DepthOfField.FarIntensity = 0
	DepthOfField.FocusDistance = 51.6
	DepthOfField.InFocusRadius = 50
	DepthOfField.NearIntensity = 6
	DepthOfField.Name = "DPT_"..gToken

	local frame = Instance.new('Frame')
	frame.Parent = Frame
	frame.Size = UDim2.new(0.95, 0, 0.95, 0)
	frame.Position = UDim2.new(0.5, 0, 0.5, 0)
	frame.AnchorPoint = Vector2.new(0.5, 0.5)
	frame.BackgroundTransparency = 1

	local GenUid; do
		local id = 0
		function GenUid()
			id = id + 1
			return 'neon::'..tostring(id)
		end
	end

	do
		local function IsNotNaN(x)
			return x == x
		end
		local continue = IsNotNaN(camera:ScreenPointToRay(0,0).Origin.x)
		while not continue do
			RunService.RenderStepped:wait()
			continue = IsNotNaN(camera:ScreenPointToRay(0,0).Origin.x)
		end
	end

	local DrawQuad; do
		local acos, max, pi, sqrt = math.acos, math.max, math.pi, math.sqrt
		local sz = 0.22
		local function DrawTriangle(v1, v2, v3, p0, p1)
			local s1 = (v1 - v2).magnitude
			local s2 = (v2 - v3).magnitude
			local s3 = (v3 - v1).magnitude
			local smax = max(s1, s2, s3)
			local A, B, C
			if s1 == smax then
				A, B, C = v1, v2, v3
			elseif s2 == smax then
				A, B, C = v2, v3, v1
			elseif s3 == smax then
				A, B, C = v3, v1, v2
			end

			local para = ( (B-A).x*(C-A).x + (B-A).y*(C-A).y + (B-A).z*(C-A).z ) / (A-B).magnitude
			local perp = sqrt((C-A).magnitude^2 - para*para)
			local dif_para = (A - B).magnitude - para

			local st = CFrame.new(B, A)
			local za = CFrame.Angles(pi/2,0,0)

			local cf0 = st
			local Top_Look = (cf0 * za).lookVector
			local Mid_Point = A + CFrame.new(A, B).lookVector * para
			local Needed_Look = CFrame.new(Mid_Point, C).lookVector
			local dot = Top_Look.x*Needed_Look.x + Top_Look.y*Needed_Look.y + Top_Look.z*Needed_Look.z

			local ac = CFrame.Angles(0, 0, acos(dot))
			cf0 = cf0 * ac
			if ((cf0 * za).lookVector - Needed_Look).magnitude > 0.01 then
				cf0 = cf0 * CFrame.Angles(0, 0, -2*acos(dot))
			end
			cf0 = cf0 * CFrame.new(0, perp/2, -(dif_para + para/2))

			local cf1 = st * ac * CFrame.Angles(0, pi, 0)
			if ((cf1 * za).lookVector - Needed_Look).magnitude > 0.01 then
				cf1 = cf1 * CFrame.Angles(0, 0, 2*acos(dot))
			end
			cf1 = cf1 * CFrame.new(0, perp/2, dif_para/2)

			if not p0 then
				p0 = Instance.new('Part')
				p0.FormFactor = 'Custom'
				p0.TopSurface = 0
				p0.BottomSurface = 0
				p0.Anchored = true
				p0.CanCollide = false
				p0.CastShadow = false
				p0.Material = MTREL
				p0.Size = Vector3.new(sz, sz, sz)
				local mesh = Instance.new('SpecialMesh', p0)
				mesh.MeshType = 2
				mesh.Name = 'WedgeMesh'
			end
			p0.WedgeMesh.Scale = Vector3.new(0, perp/sz, para/sz)
			p0.CFrame = cf0

			if not p1 then p1 = p0:clone() end
			p1.WedgeMesh.Scale = Vector3.new(0, perp/sz, dif_para/sz)
			p1.CFrame = cf1

			return p0, p1
		end

		function DrawQuad(v1, v2, v3, v4, parts)
			parts[1], parts[2] = DrawTriangle(v1, v2, v3, parts[1], parts[2])
			parts[3], parts[4] = DrawTriangle(v3, v2, v4, parts[3], parts[4])
		end
	end

	if binds[frame] then return binds[frame].parts end

	local uid = GenUid()
	local parts = {}
	local f = Instance.new('Folder', root)
	f.Name = frame.Name

	local parents = {}
	do
		local function add(child)
			if child:IsA'GuiObject' then
				parents[#parents + 1] = child
				add(child.Parent)
			end
		end
		add(frame)
	end

	local function UpdateOrientation(fetchProps)
		local properties = {
			Transparency = 0.98;
			BrickColor = BrickColor.new('Institutional white');
		}
		local zIndex = 1 - 0.05*frame.ZIndex

		local tl, br = frame.AbsolutePosition, frame.AbsolutePosition + frame.AbsoluteSize
		local tr, bl = Vector2.new(br.x, tl.y), Vector2.new(tl.x, br.y)
		do
			local rot = 0;
			for _, v in ipairs(parents) do rot = rot + v.Rotation end
			if rot ~= 0 and rot%180 ~= 0 then
				local mid = tl:lerp(br, 0.5)
				local s, c = math.sin(math.rad(rot)), math.cos(math.rad(rot))
				tl = Vector2.new(c*(tl.x - mid.x) - s*(tl.y - mid.y), s*(tl.x - mid.x) + c*(tl.y - mid.y)) + mid
				tr = Vector2.new(c*(tr.x - mid.x) - s*(tr.y - mid.y), s*(tr.x - mid.x) + c*(tr.y - mid.y)) + mid
				bl = Vector2.new(c*(bl.x - mid.x) - s*(bl.y - mid.y), s*(bl.x - mid.x) + c*(bl.y - mid.y)) + mid
				br = Vector2.new(c*(br.x - mid.x) - s*(br.y - mid.y), s*(br.x - mid.x) + c*(br.y - mid.y)) + mid
			end
		end
		DrawQuad(
			camera:ScreenPointToRay(tl.x, tl.y, zIndex).Origin, 
			camera:ScreenPointToRay(tr.x, tr.y, zIndex).Origin, 
			camera:ScreenPointToRay(bl.x, bl.y, zIndex).Origin, 
			camera:ScreenPointToRay(br.x, br.y, zIndex).Origin, 
			parts
		)
		if fetchProps then
			for _, pt in pairs(parts) do pt.Parent = f end
			for propName, propValue in pairs(properties) do
				for _, pt in pairs(parts) do pt[propName] = propValue end
			end
		end
	end

	UpdateOrientation(true)
	RunService:BindToRenderStep(uid, 2000, UpdateOrientation)
end

local function Draggable(Bar, Window, enableTaptic, tapticOffset)
	pcall(function()
		local Dragging, DragInput, MousePos, FramePos
		local function connectFunctions()
			if dragBar and enableTaptic then
				dragBar.MouseEnter:Connect(function()
					if not Dragging then
						TweenService:Create(dragBarCosmetic, TweenInfo.new(0.25, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {BackgroundTransparency = 0.5, Size = UDim2.new(0, 120, 0, 4)}):Play()
					end
				end)
				dragBar.MouseLeave:Connect(function()
					if not Dragging then
						TweenService:Create(dragBarCosmetic, TweenInfo.new(0.25, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {BackgroundTransparency = 0.7, Size = UDim2.new(0, 100, 0, 4)}):Play()
					end
				end)
			end
		end

		connectFunctions()

		Bar.InputBegan:Connect(function(Input)
			if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
				Dragging = true
				MousePos = Input.Position
				FramePos = Window.Position

				if enableTaptic then
					TweenService:Create(dragBarCosmetic, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = UDim2.new(0, 110, 0, 4), BackgroundTransparency = 0}):Play()
				end

				Input.Changed:Connect(function()
					if Input.UserInputState == Enum.UserInputState.End then
						Dragging = false
						connectFunctions()
						if enableTaptic then
							TweenService:Create(dragBarCosmetic, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = UDim2.new(0, 100, 0, 4), BackgroundTransparency = 0.7}):Play()
						end
					end
				end)
			end
		end)

		Bar.InputChanged:Connect(function(Input)
			if Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch then
				DragInput = Input
			end
		end)

		UserInputService.InputChanged:Connect(function(Input)
			if Input == DragInput and Dragging then
				local Delta = Input.Position - MousePos
				local newMainPosition = UDim2.new(FramePos.X.Scale, FramePos.X.Offset + Delta.X, FramePos.Y.Scale, FramePos.Y.Offset + Delta.Y)
				TweenService:Create(Window, TweenInfo.new(0.35, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out), {Position = newMainPosition}):Play()

				if dragBar then
					local newDragBarPosition = UDim2.new(FramePos.X.Scale, FramePos.X.Offset + Delta.X, FramePos.Y.Scale, FramePos.Y.Offset + Delta.Y + 240)
					dragBar.Position = newDragBarPosition
				end
			end
		end)
	end)
end

			if not card then return end

					if string.find(child.Name:lower(), "title") or child.Text == "Discord" or child.Text == "Friends" or child.Text == "Server" or child.Text == "Players" or child.Text == "Time" then
						child.Text = title
					elseif string.find(child.Name:lower(), "desc") or string.find(child.Name:lower(), "sub") or string.find(child.Text:lower(), "join") or string.find(child.Text:lower(), "discord") then
						if desc then child.Text = desc end
					end
				elseif child:IsA("ImageLabel") and iconId then
					child.Image = "rbxassetid://" .. tostring(iconId)
					if iconColor then
						child.ImageColor3 = iconColor
					end
				end
			end
		end

		local function updateHomeGates()
			local friendsFrame = db:FindFirstChild("Friends")
			if not friendsFrame then return end
			
			local rawGateList, _ = getGates()
			local validGates = {}
			for _, g in ipairs(rawGateList) do
				if g ~= "No Gates Found" then
					table.insert(validGates, g)
				end
			end
			
			local cards = {
				friendsFrame:FindFirstChild("Gate1"),
				friendsFrame:FindFirstChild("Gate2"),
				friendsFrame:FindFirstChild("Gate3"),
				friendsFrame:FindFirstChild("Gate4")
			}
			
			for i, card in ipairs(cards) do
				if card then
					local gateName = validGates[i]
					if gateName then
						card.Visible = true
						configureCard(card, gateName, "", nil, nil)
						pcall(function()
							if card:FindFirstChild("Value") then 
								card.Value.Text = "" 
								card.Value.Visible = false
							end
							if card:FindFirstChild("Title") then
								card.Title.TextScaled = false 
								card.Title.TextSize = 14
								card.Title.TextWrapped = true
								card.Title.Size = UDim2.new(1, -10, 1, 0)
							end
						end)
					else
						card.Visible = false
					end
				end
			end
		end

		local function format(Int)
			return string.format("%02i", Int)
		end

		local function convertToHMS(Seconds)
			local Minutes = (Seconds - Seconds%60)/60
			Seconds = Seconds - Minutes*60
			local Hours = (Minutes - Minutes%60)/60
			Minutes = Minutes - Hours*60
			return format(Hours)..":"..format(Minutes)..":"..format(Seconds)
		end

		coroutine.wrap(function()
			while task.wait(1) do
				pcall(function()
					db.Server.Time.Value.Text = convertToHMS(time())
				end)
				pcall(function()
					db.Server.Players.Value.Text = getSeasonTimerText()
				end)
				pcall(function()
					updateHomeGates()
				end)
			end
		end)()
	end

			function Section:CreateSlider(SliderSettings, Flag)
				TabPageSection.Position = UDim2.new(0,0,0,28)
				local SliderV = { IgnoreConfig = false, Class = "Slider", Settings = SliderSettings }

				SliderSettings = Kwargify({
					Name = "Slider",
					Range = {0, 200},
					Increment = 1,
					CurrentValue = 100,
					Callback = function(Value) end,
				}, SliderSettings or {})

				local SLDragging = false
				local Slider = Elements.Template.Slider:Clone()
				Slider.Name = SliderSettings.Name .. " - Slider"
				Slider.Title.Text = SliderSettings.Name
				Slider.Visible = true
				Slider.Parent = TabPageSection

				Slider.BackgroundTransparency = 1
				Slider.UIStroke.Transparency = 1
				Slider.Title.TextTransparency = 1

				TweenService:Create(Slider, TweenInfo.new(0.7, Enum.EasingStyle.Exponential), {BackgroundTransparency = 0.5}):Play()
				TweenService:Create(Slider.UIStroke, TweenInfo.new(0.7, Enum.EasingStyle.Exponential), {Transparency = 0.5}):Play()
				TweenService:Create(Slider.Title, TweenInfo.new(0.7, Enum.EasingStyle.Exponential), {TextTransparency = 0}):Play()	

				Slider.Main.Progress.Size =	UDim2.new(0, Slider.Main.AbsoluteSize.X * ((SliderSettings.CurrentValue + SliderSettings.Range[1]) / (SliderSettings.Range[2] - SliderSettings.Range[1])) > 5 and Slider.Main.AbsoluteSize.X * (SliderSettings.CurrentValue / (SliderSettings.Range[2] - SliderSettings.Range[1])) or 5, 1, 0)
				Slider.Value.Text = tostring(SliderSettings.CurrentValue)
				SliderV.CurrentValue = SliderSettings.CurrentValue

				pcall(function() SliderSettings.Callback(SliderSettings.CurrentValue) end)

				Slider.MouseEnter:Connect(function() tween(Slider.UIStroke, {Color = Color3.fromRGB(87, 84, 104)}) end)
				Slider.MouseLeave:Connect(function() tween(Slider.UIStroke, {Color = Color3.fromRGB(64,61,76)}) end)

				Slider.Interact.InputBegan:Connect(function(Input)
					if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then SLDragging = true end 
				end)
				Slider.Interact.InputEnded:Connect(function(Input) 
					if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then SLDragging = false end 
				end)

				Slider.Interact.MouseButton1Down:Connect(function()
					local Current = Slider.Main.Progress.AbsolutePosition.X + Slider.Main.Progress.AbsoluteSize.X
					local Start = Current
					local Location
					local Loop; Loop = RunService.Stepped:Connect(function()
						if SLDragging then
							Location = UserInputService:GetMouseLocation().X
							Current = Current + 0.025 * (Location - Start)

							if Location < Slider.Main.AbsolutePosition.X then
								Location = Slider.Main.AbsolutePosition.X
							elseif Location > Slider.Main.AbsolutePosition.X + Slider.Main.AbsoluteSize.X then
								Location = Slider.Main.AbsolutePosition.X + Slider.Main.AbsoluteSize.X
							end

							if Current < Slider.Main.AbsolutePosition.X + 5 then
								Current = Slider.Main.AbsolutePosition.X + 5
							elseif Current > Slider.Main.AbsolutePosition.X + Slider.Main.AbsoluteSize.X then
								Current = Slider.Main.AbsolutePosition.X + Slider.Main.AbsoluteSize.X
							end

							if Current <= Location and (Location - Start) < 0 then
								Start = Location
							elseif Current >= Location and (Location - Start) > 0 then
								Start = Location
							end
							Slider.Main.Progress.Size = UDim2.new(0, Location - Slider.Main.AbsolutePosition.X, 1, 0)
							local NewValue = SliderSettings.Range[1] + (Location - Slider.Main.AbsolutePosition.X) / Slider.Main.AbsoluteSize.X * (SliderSettings.Range[2] - SliderSettings.Range[1])
							NewValue = math.floor(NewValue / SliderSettings.Increment + 0.5) * (SliderSettings.Increment * 10000000) / 10000000
							Slider.Value.Text = tostring(NewValue)

							if SliderSettings.CurrentValue ~= NewValue then
								pcall(function() SliderSettings.Callback(NewValue) end)
								SliderSettings.CurrentValue = NewValue
								SliderV.CurrentValue = NewValue
							end
						else
							TweenService:Create(Slider.Main.Progress, TweenInfo.new(0.1, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Size = UDim2.new(0, Location - Slider.Main.AbsolutePosition.X > 5 and Location - Slider.Main.AbsolutePosition.X or 5, 1, 0)}):Play()
							Loop:Disconnect()
						end
					end)
				end)

				local function Set(NewVal, bleh)
					NewVal = NewVal or SliderSettings.CurrentValue
					TweenService:Create(Slider.Main.Progress, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.InOut), {Size = UDim2.new(0, Slider.Main.AbsoluteSize.X * ((NewVal + SliderSettings.Range[1]) / (SliderSettings.Range[2] - SliderSettings.Range[1])) > 5 and Slider.Main.AbsoluteSize.X * (NewVal / (SliderSettings.Range[2] - SliderSettings.Range[1])) or 5, 1, 0)}):Play()
					if not bleh then Slider.Value.Text = tostring(NewVal) end
					pcall(function() SliderSettings.Callback(NewVal) end)
					SliderSettings.CurrentValue = NewVal
					SliderV.CurrentValue = NewVal
				end

				function SliderV:UpdateValue(Value) Set(tonumber(Value)) end 

				Slider.Value:GetPropertyChangedSignal("Text"):Connect(function()
					local text = Slider.Value.Text
					if not tonumber(text) and text ~= "." then
						Slider.Value.Text = text:match("[0-9.]*") or ""
					end
					if SliderSettings.Range[2] < (tonumber(Slider.Value.Text) or 0) then Slider.Value.Text = SliderSettings.Range[2] end
					Slider.Value.Size = UDim2.fromOffset(Slider.Value.TextBounds.X, 23)
					Set(tonumber(Slider.Value.Text), true)
				end)

				function SliderV:Set(NewSliderSettings)
					NewSliderSettings = Kwargify(SliderSettings, NewSliderSettings or {})
					SliderSettings = NewSliderSettings
					SliderV.Settings = NewSliderSettings
					Slider.Name = SliderSettings.Name .. " - Slider"
					Slider.Title.Text = SliderSettings.Name
					Set()
				end

				function SliderV:Destroy()
					Slider.Visible = false
					Slider:Destroy()
				end

				if Flag then Luna.Options[Flag] = SliderV end
				return SliderV
			end

					if Dropdown.List:FindFirstChild(name) then

				local function Refresh()
					Clear()
					for i,v in pairs(DropdownSettings.Options) do
						local Option = Dropdown.List.Template:Clone()
						local optionhover = false
						Option.Text = v
						Option.Name = v
						Option.Interact.MouseButton1Click:Connect(function()
							local bleh
							if DropdownSettings.MultipleOptions then
								if table.find(DropdownSettings.CurrentOption, v) then
									RemoveTable(DropdownSettings.CurrentOption, v)
									DropdownV.CurrentOption = DropdownSettings.CurrentOption
									if not optionhover then tween(Option, {TextColor3 = Color3.fromRGB(200,200,200)}) end
									tween(Option, {BackgroundTransparency = 0.98})
								else
									table.insert(DropdownSettings.CurrentOption, v)
									DropdownV.CurrentOption = DropdownSettings.CurrentOption
									tween(Option, {TextColor3 = Color3.fromRGB(240,240,240), BackgroundTransparency = 0.95})
								end
								bleh = DropdownSettings.CurrentOption
							else
								DropdownSettings.CurrentOption = {v}
								bleh = v
								DropdownV.CurrentOption = bleh
								ActivateColorSingle(v)
							end

							SafeCallback(bleh, function()
								if DropdownSettings.MultipleOptions then
									if DropdownSettings.CurrentOption and type(DropdownSettings.CurrentOption) == "table" then
										Dropdown.Selected.PlaceholderText = #DropdownSettings.CurrentOption == 1 and DropdownSettings.CurrentOption[1] or #DropdownSettings.CurrentOption == 0 and "None" or unpackt(DropdownSettings.CurrentOption)
									else

										Dropdown.Selected.PlaceholderText = "None"
									end
								else
									Dropdown.Selected.PlaceholderText = DropdownSettings.CurrentOption[1] or "None"
								end
								Dropdown.Selected.Text = ""
							end)
						end)
						Option.Visible = true
						Option.Parent = Dropdown.List
						Option.MouseEnter:Connect(function()
							optionhover = true
							if Option.BackgroundTransparency ~= 0.95 then tween(Option, {TextColor3 = Color3.fromRGB(240,240,240)}) end
						end)
						Option.MouseLeave:Connect(function()
							optionhover = false
							if Option.BackgroundTransparency ~= 0.95 then tween(Option, {TextColor3 = Color3.fromRGB(200,200,200)}) end
						end)	
					end

				Dropdown.MouseLeave:Connect(function() tween(Dropdown.UIStroke, {Color = Color3.fromRGB(64,61,76)}) end)

				Refresh()

				if DropdownSettings.CurrentOption then
					if type(DropdownSettings.CurrentOption) == "string" then DropdownSettings.CurrentOption = {DropdownSettings.CurrentOption} end
				else
					DropdownSettings.CurrentOption = {}
				end

				function DropdownV:Set(NewDropdownSettings)
					NewDropdownSettings = Kwargify(DropdownSettings, NewDropdownSettings or {})
					DropdownV.Settings = NewDropdownSettings
					DropdownSettings = NewDropdownSettings
					Dropdown.Name = DropdownSettings.Name
					Dropdown.Title.Text = DropdownSettings.Name
					Refresh()

					if DropdownSettings.MultipleOptions then
						Dropdown.Selected.PlaceholderText = #DropdownSettings.CurrentOption == 1 and DropdownSettings.CurrentOption[1] or #DropdownSettings.CurrentOption == 0 and "None" or unpackt(DropdownSettings.CurrentOption)
					else
						Dropdown.Selected.PlaceholderText = DropdownSettings.CurrentOption[1] or "None"
					end
					Dropdown.Selected.Text = ""
				end

					Dropdown.Visible = false
					Dropdown:Destroy()

	local lastHealFire = tick()

					gui.Parkour.Script.iFrame:FireServer()
				end
			end)
		end

			pcall(function()
				local mobsFolder = workspace:FindFirstChild("Mobs")
				if mobsFolder then
					for _, mob in pairs(mobsFolder:GetChildren()) do
						if _G.AutoAllMobs then
							if mob:FindFirstChild("Stunned") then
								mob.Stunned.Value = true
							end
						elseif _G.SlingshotElven or _G.SlingshotStatue or _G.AutoDungeon then
							if (mob.Name == "Elven Sorceress" or mob.Name == "Statue") and mob:FindFirstChild("Stunned") then
								mob.Stunned.Value = true
							end
						end
					end
				end
			end)
		end

		if combatActive then
			if tick() - lastConsumableFire >= 1 then
				pcall(function()
					local remotes = ReplicatedStorage:FindFirstChild("Remotes")
					if remotes then
						if remotes:FindFirstChild("IchorChalice") then
							remotes.IchorChalice:FireServer()
						end
						if remotes:FindFirstChild("Ambrosia") then
							remotes.Ambrosia:FireServer()
						end
						if remotes:FindFirstChild("AethericExtract") then
							remotes.AethericExtract:FireServer()
						end
					end
				end)
				lastConsumableFire = tick()
			end

			pcall(function()
				local char = Player.Character
				if char and char:FindFirstChild("Health") and char:FindFirstChild("MaxHealth") then
					local h = char.Health.Value
					local m = char.MaxHealth.Value
					
					if m > 0 and (h / m) < 0.50 then
						if tick() - lastHealFire >= 1 then
							lastHealFire = tick()
							
							task.spawn(function()
								local backpack = Player:FindFirstChild("Backpack")

								if backpack then pcall(function() backpack.CureMeadow.ServerScript.Charge:FireServer() end) end
								task.wait(0.3)
								
								pcall(function() Player.PlayerGui.Parkour.Script.Healer:FireServer() end)
								task.wait(0.3)
								
								if backpack then pcall(function() backpack.LatricReclamation.ServerScript.Charge:FireServer() end) end
							end)
						end
					end
				end
			end)
		end
	end
end)

-- SERVER CO-PRESENCE FAILSAFE
-- ==========================================
local function executeFailsafeTeleport()
	_G.AutoDungeon = false
	_G.AutoKingsRuins = false
	_G.SlingshotStatue = false
	_G.SlingshotElven = false
	_G.SlingshotMimic = false
	_G.AutoAllMobsOverworld = false
	_G.AutoAllMobs = false

	CancelSmoothTween()
	toggleNoclip(false)
	resetWalkSpeed()
	
	updateUIRunsStatus("Failsafe: Other players detected!")
	sendWebhookNotification("Failsafe activated: another player detected in server.")

	pcall(function()
		local char = Player.Character
		local hrp = char and char:FindFirstChild("HumanoidRootPart")
		if hrp then
			if _G.SavedOriginalPosition then
				hrp.CFrame = _G.SavedOriginalPosition
			else
				local spFolder = workspace:FindFirstChild("SpawnPoints")
				local fallbackSpawn = spFolder and spFolder:FindFirstChildWhichIsA("BasePart")
				if fallbackSpawn then
					hrp.CFrame = fallbackSpawn.CFrame + Vector3.new(0, 5, 0)
				end
			end
			stopMomentum(hrp)
		end
	end)

	task.wait(1.5)
	updateUIRunsStatus("Failsafe: Server hopping...")
	executeProjectedServerHop()
end

		if (_G.AutoKingsRuins or _G.AutoDungeon) then

					if not isPlayerAuthorized(player.Name) then
						unauthorizedPlayerFound = true
						break
					end
				end
				
				if unauthorizedPlayerFound then
					Luna:Notification({
						Title = "Failsafe Triggered",
						Content = "Another player joined your dungeon. Teleporting to safety...",
						Icon = "warning"
					})
					executeFailsafeTeleport()
				end
			end
		end
	end
end)

local function SlingshotKillAll(toggleKey)
	task.spawn(function()
		saveOriginalPosition()

		local PULLBACK_DISTANCE = 50
		local DASH_TIME = 0
		local LOOP_DELAY = 0
		local FIRE_PERCENT = 1
		local FAKE_SPEED = 3000

		local function fireODMRemotes(myRoot, targetRoot)
			local playerGui = Player:FindFirstChild("PlayerGui")
			if playerGui then pcall(function() playerGui.Parkour.Script.ODMSlash:FireServer("Right") end) end

			local char = Player.Character
			if char and char:FindFirstChild("TitanBlades") then
				pcall(function()
					local ss = char.TitanBlades:FindFirstChild("SwordScript")
					if ss and ss:FindFirstChild("ODMDown") then ss.ODMDown:FireServer() end
					if ss and ss:FindFirstChild("ODMUp") then ss.ODMUp:FireServer() end
				end)
			end

			local direction = (targetRoot.Position - myRoot.Position).Unit
			myRoot.AssemblyLinearVelocity = direction * FAKE_SPEED
		end

		while _G[toggleKey] do
			local myChar = Player.Character
			local targetMob = nil
			local mobsFolder = workspace:FindFirstChild("Mobs")

			if mobsFolder and myChar and myChar:FindFirstChild("HumanoidRootPart") then
				local myRoot = myChar.HumanoidRootPart
				local shortestDist = math.huge
				for _, v in pairs(mobsFolder:GetChildren()) do
					if v.Name ~= "Regulus" and v:FindFirstChild("HumanoidRootPart") and isEntityAlive(v) then
						local dist = (v.HumanoidRootPart.Position - myRoot.Position).Magnitude
						if dist < shortestDist then
							shortestDist = dist
							targetMob = v
						end
					end
				end
			end

			if targetMob then
				local cam = workspace.CurrentCamera
				local camConn
				if cam then
					cam.CameraType = Enum.CameraType.Scriptable
					camConn = RunService.RenderStepped:Connect(function()
						if targetMob and targetMob:FindFirstChild("HumanoidRootPart") then
							local tPos = targetMob.HumanoidRootPart.Position
							cam.CFrame = CFrame.lookAt(tPos + Vector3.new(0, 30, 40), tPos)
						end
					end)
				end

				local lastHRFire = 0
				local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")

				while _G[toggleKey] and isEntityAlive(targetMob) and myChar and myChar:FindFirstChild("HumanoidRootPart") do
					myRoot = myChar.HumanoidRootPart
					if tick() - lastHRFire >= 1 then
						triggerHeavenlyRestriction()
						lastHRFire = tick()
					end

					local targetRoot = targetMob:FindFirstChild("HumanoidRootPart")
					if not targetRoot then break end

					myRoot.CFrame = targetRoot.CFrame * CFrame.new(0, PULLBACK_DISTANCE, PULLBACK_DISTANCE)

						end
					end)

					dashTween.Completed:Wait()

				end

				if camConn then camConn:Disconnect() end

local function SlingshotKillOverworld(toggleKey)
	task.spawn(function()
		saveOriginalPosition()
		local safeCFrame = _G.SavedOriginalPosition

		local PULLBACK_DISTANCE = 100
		local DASH_TIME = 0
		local LOOP_DELAY = 0
		local FIRE_PERCENT = 1
		local FAKE_SPEED = 3000

		local mobAges = {}
		local lastKillTime = tick()

		local function fireODMRemotes(myRoot, targetRoot)
			local playerGui = Player:FindFirstChild("PlayerGui")
			if playerGui then pcall(function() playerGui.Parkour.Script.ODMSlash:FireServer("Right") end) end

			local char = Player.Character
			if char and char:FindFirstChild("TitanBlades") then
				pcall(function()
					local ss = char.TitanBlades:FindFirstChild("SwordScript")
					if ss and ss:FindFirstChild("ODMDown") then ss.ODMDown:FireServer() end
					if ss and ss:FindFirstChild("ODMUp") then ss.ODMUp:FireServer() end
				end)
			end

			local direction = (targetRoot.Position - myRoot.Position).Unit
			myRoot.AssemblyLinearVelocity = direction * FAKE_SPEED
		end

		while _G[toggleKey] do
			local myChar = Player.Character
			local targetMob = nil
			local mobsFolder = workspace:FindFirstChild("Mobs")
			local effects = workspace:FindFirstChild("Effects")

			local regulus = mobsFolder and mobsFolder:FindFirstChild("Regulus")
			if regulus and isEntityAlive(regulus) then
				task.spawn(function()
					if _G.ToggleAllMobs then
						_G.ToggleAllMobs:UpdateState(false)
						_G.ToggleAllMobs.Settings.Callback(false) 
					end
					
					task.wait(0.5)

					unequipCurrent()
					task.wait(0.15)
					ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("BackWeapon"):FireServer("RightWeapon", "LightningAxe", "BackRight", "LightningAxe")
					task.wait(0.15)
					loadWeapons()
					task.wait(0.5)
					
					local axe = (Player.Character and Player.Character:FindFirstChild("LightningAxe")) or (Player:FindFirstChild("Backpack") and Player.Backpack:FindFirstChild("LightningAxe"))
					
					if axe and Player.Character and Player.Character:FindFirstChild("Humanoid") then
						Player.Character.Humanoid:EquipTool(axe)
					else
						VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Four, false, game)
						task.wait(0.1)
						VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Four, false, game)
					end
					task.wait(0.5)

					if _G.ToggleRegulus then
						_G.ToggleRegulus:UpdateState(true)
						_G.ToggleRegulus.Settings.Callback(true)
					end

					while _G.AutoRegulus and regulus:IsDescendantOf(workspace) and isEntityAlive(regulus) do
						task.wait(0.5)
					end

					if _G.ToggleRegulus then
						_G.ToggleRegulus:UpdateState(false)
						_G.ToggleRegulus.Settings.Callback(false)
					end

					task.wait(0.5)

					if _G.ToggleAllMobs then
						_G.ToggleAllMobs:UpdateState(true)
						_G.ToggleAllMobs.Settings.Callback(true)
					end
				end)
				return 
			end

			if mobsFolder and myChar and myChar:FindFirstChild("HumanoidRootPart") then
				local myRoot = myChar.HumanoidRootPart
				local validMobs = {}

				for m, _ in pairs(mobAges) do
					if not m:IsDescendantOf(workspace) then
						mobAges[m] = nil
					end
				end

				for _, v in pairs(mobsFolder:GetChildren()) do
					if v.Name ~= "Regulus" and v:FindFirstChild("HumanoidRootPart") and isEntityAlive(v) then
						if not mobAges[v] then
							mobAges[v] = tick()
						end
						table.insert(validMobs, {mob = v, age = mobAges[v]})
					end
				end

				local shouldThrottle = true
				local isGateFinished = false
				
				if effects then
					local worldType = effects:FindFirstChild("WorldType")
					local gate = effects:FindFirstChild("Gate")
					
					if worldType and worldType.Value == "Gate" and gate then
						local totalMobs = gate:FindFirstChild("TotalMobs")
						local mobsDefeated = gate:FindFirstChild("MobsDefeated")
						
						if totalMobs and mobsDefeated then
							local remainingToDefeat = totalMobs.Value - mobsDefeated.Value
							
							if remainingToDefeat <= #validMobs then
								shouldThrottle = false
							end
							
							if totalMobs.Value < 5 then
								shouldThrottle = false
							end
							
							if remainingToDefeat <= 0 then
								isGateFinished = true
							end
						end
					end
				end

				if #validMobs > 0 and not isGateFinished then
					if shouldThrottle then
						if #validMobs >= 2 then
							table.sort(validMobs, function(a, b) return a.age < b.age end)
							targetMob = validMobs[1].mob
						else
							targetMob = nil 
						end
					else
						table.sort(validMobs, function(a, b) return a.age < b.age end)
						targetMob = validMobs[1].mob
					end
				end
			end

			if targetMob then
				local cam = workspace.CurrentCamera
				local camConn
				if cam then
					cam.CameraType = Enum.CameraType.Scriptable
					camConn = RunService.RenderStepped:Connect(function()
						if targetMob and targetMob:FindFirstChild("HumanoidRootPart") then
							local tPos = targetMob.HumanoidRootPart.Position
							cam.CFrame = CFrame.lookAt(tPos + Vector3.new(0, 30, 40), tPos)
						end
					end)
				end

				local lastHRFire = 0
				local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")

				while _G[toggleKey] and isEntityAlive(targetMob) and myChar and myChar:FindFirstChild("HumanoidRootPart") do
					myRoot = myChar.HumanoidRootPart
					if tick() - lastHRFire >= 1 then
						triggerHeavenlyRestriction()
						lastHRFire = tick()
					end

					local targetRoot = targetMob:FindFirstChild("HumanoidRootPart")
					if not targetRoot then break end

					myRoot.CFrame = targetRoot.CFrame * CFrame.new(0, PULLBACK_DISTANCE, PULLBACK_DISTANCE)
					local strikeCFrame = targetRoot.CFrame * CFrame.new(0, 0, -3)
					local dashTween = TweenService:Create(myRoot, TweenInfo.new(DASH_TIME, Enum.EasingStyle.Linear), {CFrame = strikeCFrame})

					dashTween:Play()

					task.delay(DASH_TIME * FIRE_PERCENT, function()
						if _G[toggleKey] and myRoot and targetRoot and isEntityAlive(targetMob) then
							pcall(function() fireODMRemotes(myRoot, targetRoot) end)
						end
					end)

					dashTween.Completed:Wait()
					stopMomentum(myRoot)
					task.wait(LOOP_DELAY)
				end

				stopMomentum(myRoot)
				if camConn then camConn:Disconnect() end
				if cam then cam.CameraType = Enum.CameraType.Custom end
				
				lastKillTime = tick()
			else
				local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
				if myRoot then
					stopMomentum(myRoot)
					if safeCFrame then
						myRoot.CFrame = safeCFrame
					end
				end
			end
			task.wait(0.1)
		end

		local finalCam = workspace.CurrentCamera
		if finalCam then finalCam.CameraType = Enum.CameraType.Custom end
		restoreOriginalPosition()
	end)
end

local function TreasureChestFarm()
	task.spawn(function()
		while _G.AutoTreasure do
			local chestsFolder = workspace:FindFirstChild("Chests")
			local targetChest = nil
			local shortestDist = math.huge
			local char = Player.Character
			local hrp = char and char:FindFirstChild("HumanoidRootPart")

			if chestsFolder and hrp then
				for _, chest in pairs(chestsFolder:GetChildren()) do
					if (chest.Name == "TreasureChest" or chest.Name == "SmallChest") and chest:FindFirstChild("Center") then
						local prompt = chest.Center:FindFirstChildWhichIsA("ProximityPrompt")
						if prompt and prompt.Enabled then
							local dist = (chest.Center.Position - hrp.Position).Magnitude
							if dist < shortestDist then
								shortestDist = dist
								targetChest = chest
							end
						end
					end
				end
			end

			if targetChest then
				local center = targetChest:FindFirstChild("Center")
				if hrp and center then
					hrp.CFrame = center.CFrame
					stopMomentum(hrp)
					local prompt = center:FindFirstChildWhichIsA("ProximityPrompt")
					if prompt then
						prompt.HoldDuration = 0
						task.wait(0.1)
						pcall(function() fireproximityprompt(prompt) end)
						task.wait(0.5)
					end
				end
			else
				_G.AutoTreasure = false
			end
			task.wait(0.1)
		end
	end)
end

local function getPickaxe(character)
	for _, child in pairs(character:GetChildren()) do
		if child:IsA("Tool") and string.find(string.lower(child.Name), "pickaxe") then
			return child
		end
	end
	local backpack = Player:FindFirstChild("Backpack")
	if backpack then
		for _, child in pairs(backpack:GetChildren()) do
			if child:IsA("Tool") and string.find(string.lower(child.Name), "pickaxe") then
				return child
			end
		end
	end
	return nil
end

local isMining = false
local function GrandOreFarm()
	task.spawn(function()
		task.spawn(function()
			while _G.AutoGrandOre do
				pcall(function()
					local char = Player.Character
					local healthValue = char and char:FindFirstChild("Health")
					local maxHealthValue = char and char:FindFirstChild("MaxHealth")
					
					if isMining and healthValue and maxHealthValue then
						local hp = healthValue.Value
						local maxHp = maxHealthValue.Value
						
						if hp < maxHp then
							local backpack = Player:FindFirstChild("Backpack")
							pcall(function() ReplicatedStorage.Remotes.Ambrosia:FireServer() end)
							pcall(function() ReplicatedStorage.Remotes.IchorChalice:FireServer() end)
							if backpack then
								pcall(function() backpack.CureMeadow.ServerScript.Charge:FireServer() end)
								pcall(function() backpack.LatricReclamation.ServerScript.Charge:FireServer() end)
							end
						end
						
						if (hp / maxHp) >= 0.40 then
							local playerGui = Player:FindFirstChild("PlayerGui")
							local frenzyRemote = playerGui and playerGui:FindFirstChild("Parkour") and playerGui.Parkour:FindFirstChild("Script") and playerGui.Parkour.Script:FindFirstChild("FrenzyPickaxe")
							
							if frenzyRemote then
								frenzyRemote:FireServer()
							end
						end
					end
				end)
				task.wait(1)
			end
		end)

		while _G.AutoGrandOre do 
			local foundOre = false
			pcall(function()
				local character = Player.Character
				if not character then return end
				local hrp = character:FindFirstChild("HumanoidRootPart")
				if not hrp then return end
				
				local oresFolder = workspace:FindFirstChild("Ores")
				if not oresFolder then return end

				local closestOre = nil
				local shortestDistance = math.huge 

				for i, v in pairs(oresFolder:GetChildren()) do
					local mainPart = v:FindFirstChild("Main")
					if mainPart and mainPart:IsA("BasePart") then
						local hpValue = v:FindFirstChild("HP")
						local isAlive = not hpValue or (hpValue and hpValue.Value > 0)
						if isAlive then
							local distance = (hrp.Position - mainPart.Position).Magnitude
							if distance < shortestDistance then
								shortestDistance = distance
								closestOre = v
							end
						end
					end
				end

				if closestOre then
					foundOre = true
					if character and not character:FindFirstChild("FrenzyPickaxe") then
						unequipCurrent()
						task.wait(0.15)
						
						local args = {
							"RightWeapon",
							"FrenzyPickaxe",
							"BackRight",
							"FrenzyPickaxe"
						}
						pcall(function() ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("BackWeapon"):FireServer(unpack(args)) end)
						
						task.wait(0.15)
						loadWeapons()
						task.wait(0.5) 
						
						if character and not character:FindFirstChild("FrenzyPickaxe") then
							VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Four, false, game)
							VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Four, false, game)
							task.wait(0.5)
						end
					end
					
					local targetPart = closestOre.Main
					local hpValue = closestOre:FindFirstChild("HP")
					isMining = false
					
					local flatHRP = Vector3.new(hrp.Position.X, targetPart.Position.Y, hrp.Position.Z)
					local dir = (flatHRP - targetPart.Position).Unit
					if dir.Magnitude ~= dir.Magnitude then dir = Vector3.new(1,0,0) end
					local standPosition = targetPart.Position + (dir * 3.5)
					local finalCFrame = CFrame.lookAt(standPosition, targetPart.Position)
					
					local success = SmoothTween(finalCFrame, 100, function() return _G.AutoGrandOre end)
					if success then
						isMining = true 
						local lastHP = hpValue and hpValue.Value or 100
						local stuckTimer = tick()

						repeat 
							if hrp and targetPart and (hrp.Position - targetPart.Position).Magnitude <= 15 then
								local pickaxe = getPickaxe(character)
								if pickaxe then
									local scriptObj = pickaxe:FindFirstChild("SwordScript")
									if scriptObj and scriptObj:FindFirstChild("Activate") then
										scriptObj.Activate:FireServer()
									end
								end
							end
							
							task.wait(0.1) 
							
							if tick() - stuckTimer >= 2 then
								if hpValue and hpValue.Value >= lastHP then
									break
								else
									lastHP = hpValue.Value
									stuckTimer = tick()
								end
							end
						until not closestOre:IsDescendantOf(workspace) or (hpValue and hpValue.Value <= 0) or not _G.AutoGrandOre
						isMining = false
					end
				end
			end)

			if not foundOre then
				_G.AutoGrandOre = false
			end
			task.wait(0.1)
		end
		CancelSmoothTween()
		isMining = false
	end)
end

				end
			end)
		end
	end)
end

local function getFurthestPart(roomModel, rootPosition)
	local furthestPart = nil
	local maxDist = 0
	for _, obj in pairs(roomModel:GetDescendants()) do
		if obj:IsA("BasePart") then
			local dist = (obj.Position - rootPosition).Magnitude
			if dist > maxDist and dist < 200 then
				maxDist = dist
				furthestPart = obj
			end
		end
	end
	return furthestPart
end

local function getActiveFloor(genMap, root)
	local closestFloor = nil
	local minDist = math.huge
	for i = 1, 3 do
		local f = genMap:FindFirstChild("Floor"..i)
		if f then
			for _, room in pairs(f:GetChildren()) do
				if room:IsA("Model") and room:GetPivot() then
					local d = (room:GetPivot().Position - root.Position).Magnitude
					if d < minDist then
						minDist = d
						closestFloor = f
					end
				end
			end
		end
	end
	return closestFloor
end

local function getIncompleteEventRoom(floorModel)
	if not floorModel then return nil end
	for _, room in pairs(floorModel:GetChildren()) do
		if room.Name == "EventRoom" then
			local comp = room:FindFirstChild("Complete")
			if comp and comp.Value == false then
				return room
			end
		end
	end
	return nil
end

local function isExitRoomReady(floorModel)
	if not floorModel then return false end
	local exitRoom = floorModel:FindFirstChild("ExitRoom")
	if not exitRoom then return false end

	local eventCount = 0
	local completeCount = 0

	for _, child in ipairs(exitRoom:GetChildren()) do
		if string.match(child.Name, "Event%d+") then
			eventCount = eventCount + 1
			local comp = child:FindFirstChild("Complete")
			if comp and comp.Value == true then
				completeCount = completeCount + 1
			end
		end
	end
	return eventCount > 0 and eventCount == completeCount
end

local function teleportToExitGate()
	local targetGate = nil
	for _, obj in pairs(workspace:GetDescendants()) do
		if obj.Name == "ExitGate" then
			targetGate = obj
			break
		end
	end
	
	if targetGate then
		local pos = targetGate:IsA("Model") and targetGate:GetPivot().Position or targetGate.Position
		local hrp = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
		if hrp then
			hrp.CFrame = CFrame.new(pos) * (hrp.CFrame - hrp.CFrame.Position)
			Luna:Notification({Title = "Kings Ruins", Content = "Transitioning to ExitGate.", Icon = "check_circle_outline"})
		end
	end
end

-- Active hostiles and chest validation checks
local function isDungeonClear()
	local mobsFolder = workspace:FindFirstChild("Mobs")
	local chestsFolder = workspace:FindFirstChild("Chests")
	
	local activeMobs = 0
	if mobsFolder then
		for _, mob in pairs(mobsFolder:GetChildren()) do
			if mob:IsA("Model") and isEntityAlive(mob) then
				if mob.Name ~= "DungeonMimicChest" then
					activeMobs = activeMobs + 1
				end
			end
		end
	end
	
	local activeChests = 0
	if chestsFolder then
		for _, chest in pairs(chestsFolder:GetChildren()) do
			if chest.Name == "TreasureChest" or chest.Name == "SmallChest" then
				local prompt = chest:FindFirstChildWhichIsA("ProximityPrompt", true)
				if prompt and prompt.Enabled then
					activeChests = activeChests + 1
				end
			end
		end
	end
	
	return (activeMobs == 0 and activeChests == 0)
end

local function startExitCheckWatcher()
	task.spawn(function()
		while _G.AutoDungeon do
			local genMap = workspace:FindFirstChild("GeneratedMap")
			local char = Player.Character
			local root = char and char:FindFirstChild("HumanoidRootPart")
			
			if genMap and root then
				local activeFloor = getActiveFloor(genMap, root)
				if activeFloor and activeFloor.Name == "Floor3" then
					local arena = activeFloor:FindFirstChild("KingsRuinsArena")
					if arena then
						local door = arena:FindFirstChild("Doors") and arena.Doors:FindFirstChild("DoorMain")
						local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)
						local doorOpened = prompt and not prompt.Enabled
						
						if doorOpened and isDungeonClear() then
							_G.KingsRuinsCompletedCount = _G.KingsRuinsCompletedCount + 1
							saveConfig()
							sendWebhookNotification("Kings Ruins Dungeon Completed Successfully!")
							teleportToExitGate()
							break
						end
					end
				end
			end
			task.wait(1)
		end
	end)
end

local function StartMasterAutoDungeon()
	task.spawn(function()
		local state = "EVALUATE"
		local currentRoom = nil
		local roomState = "INIT"
		local lastRoom = nil
		local currentTween = nil
		local lastTrapClear = 0
		local spawnTimer = 0

		startExitCheckWatcher()

		while _G.AutoDungeon do
			if tick() - lastTrapClear > 1 then
				lastTrapClear = tick()
				pcall(function()
					for _, obj in pairs(workspace.GeneratedMap:GetDescendants()) do
						if obj.Name == "TrapStorage" then obj:Destroy() end
					end
				end)
			end

			local char = Player.Character
			local root = char and char:FindFirstChild("HumanoidRootPart")
			local genMap = workspace:FindFirstChild("GeneratedMap")

			if not genMap or not root then task.wait(0.5) continue end

			if state == "EVALUATE" then
				updateUIRunsStatus("Evaluating Active Floor...")
				local activeFloor = getActiveFloor(genMap, root)
				if activeFloor then
					if activeFloor.Name == "Floor3" then
						local arena = activeFloor:FindFirstChild("KingsRuinsArena")
						if arena then
							local bossAlive = false
							local mobsFolder = workspace:FindFirstChild("Mobs")
							if mobsFolder then
								for _, mob in pairs(mobsFolder:GetChildren()) do
									if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then
										bossAlive = true
										break
									end
								end
							end
							
							local door = arena:FindFirstChild("Doors") and arena.Doors:FindFirstChild("DoorMain")
							local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)
							local doorOpened = prompt and not prompt.Enabled
							
							if doorOpened and not bossAlive then
								currentRoom = arena
								state = "HANDLE_LOOTING"
							else
								currentRoom = arena
								state = "HANDLE_FLOOR_3"
							end
						end
					else
						local room = getIncompleteEventRoom(activeFloor)
						if room then
							currentRoom = room
							state = "HANDLE_EVENT_ROOM"
						elseif isExitRoomReady(activeFloor) then
							currentRoom = activeFloor:FindFirstChild("ExitRoom")
							state = "HANDLE_EXIT_ROOM"
						end
					end
				end
				
				if currentRoom ~= lastRoom then
					roomState = "INIT"
					lastRoom = currentRoom
				end

			elseif state == "HANDLE_EVENT_ROOM" then
				local comp = currentRoom:FindFirstChild("Complete")
				if comp and comp.Value == true then
					_G.SlingshotStatue = false
					if currentTween then currentTween:Cancel(); currentTween = nil end
					preventFling(currentRoom:GetPivot() + Vector3.new(0,3,0))
					state = "EVALUATE"
					continue
				end

				local doorsModel = currentRoom:FindFirstChild("Doors")
				local door = doorsModel and doorsModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState ~= "INIT" and roomState ~= "OPEN_DOOR" and prompt and prompt.Enabled then
					if currentTween then currentTween:Cancel(); currentTween = nil end
					_G.SlingshotStatue = false
					roomState = "INIT"
					continue
				end

				if roomState == "INIT" then
					updateUIRunsStatus("Moving to event room gate...")
					if not door then
						if doorsModel and doorsModel:GetPivot() then
							preventFling(doorsModel:GetPivot() + Vector3.new(0, 5, 0))
						else
							preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0))
						end
						task.wait(1)
					else
						if (door.Position - root.Position).Magnitude > 15 then
							preventFling(door.CFrame * CFrame.new(0, 0, 6))
							task.wait(1)
						else
							roomState = "OPEN_DOOR"
						end
					end

				elseif roomState == "OPEN_DOOR" then
					updateUIRunsStatus("Opening room gate...")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0
						pcall(function() fireproximityprompt(prompt) end)
						task.wait(1)
					elseif prompt and not prompt.Enabled then
						roomState = "WALK_IN"
					end

				elseif roomState == "WALK_IN" then
					updateUIRunsStatus("Advancing into room...")
					local closeSound = currentRoom:FindFirstChild("Doors") and currentRoom.Doors:FindFirstChild("WallBlock") and currentRoom.Doors.WallBlock:FindFirstChild("Close")
					
					if closeSound and closeSound.Playing then
						if currentTween then currentTween:Cancel(); currentTween = nil end
						toggleNoclip(false)
						preventFling() 
						roomState = "WAIT_CLOSE"
					else
						if not currentTween or currentTween.PlaybackState ~= Enum.PlaybackState.Playing then
							local furthestPart = getFurthestPart(currentRoom, root.Position)
							local targetPos = furthestPart and furthestPart.Position or currentRoom:GetPivot().Position
							
							targetPos = Vector3.new(targetPos.X, currentRoom:GetPivot().Position.Y + 3, targetPos.Z)
							
							toggleNoclip(true)
							local dist = (root.Position - targetPos).Magnitude
							local tInfo = TweenInfo.new(dist / 70, Enum.EasingStyle.Linear)
							currentTween = TweenService:Create(root, tInfo, {CFrame = CFrame.new(targetPos)})
							currentTween:Play()
						end
					end

				elseif roomState == "WAIT_CLOSE" then
					updateUIRunsStatus("Waiting for entry gate to lock...")
					local closeSound = currentRoom:FindFirstChild("Doors") and currentRoom.Doors:FindFirstChild("WallBlock") and currentRoom.Doors.WallBlock:FindFirstChild("Close")
					if closeSound and not closeSound.Playing then
						roomState = "COMBAT"
					elseif not closeSound then
						roomState = "COMBAT"
					end

				elseif roomState == "COMBAT" then
					updateUIRunsStatus("Clearing event guardians...")
					if not _G.SlingshotStatue then
						_G.SlingshotStatue = true
						SlingshotKill("Statue", "SlingshotStatue")
					end
				end

			elseif state == "HANDLE_EXIT_ROOM" then
				local gateModel = currentRoom:FindFirstChild("Gate")
				local door = gateModel and gateModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState == "INIT" then
					updateUIRunsStatus("Navigating to exit gate...")
					if not door then
						if gateModel and gateModel:GetPivot() then
							preventFling(gateModel:GetPivot() + Vector3.new(0, 5, 0))
						else
							preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0))
						end
						task.wait(1)
					else
						if (door.Position - root.Position).Magnitude > 15 then
							preventFling(door.CFrame * CFrame.new(0, 0, 8))
							task.wait(1)
						else
							roomState = "OPEN_GATE"
						end
					end
				
				elseif roomState == "OPEN_GATE" then
					updateUIRunsStatus("Opening floor exit portal...")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0
						pcall(function() fireproximityprompt(prompt) end)
						task.wait(1)
					elseif prompt and not prompt.Enabled then
						roomState = "WALK_THROUGH"
					end

				elseif roomState == "WALK_THROUGH" then
					updateUIRunsStatus("Entering next floor portal...")
					if door then
						local nextStart = currentRoom:FindFirstChild("NextFloorStart")
						if nextStart then
							preventFling(nextStart.CFrame + Vector3.new(0, 5, 0))
						else
							preventFling(door.CFrame * CFrame.new(0, 0, -25))
						end
						task.wait(3.5)
						roomState = "WAIT_FLOOR_LOAD"
					end

				elseif roomState == "WAIT_FLOOR_LOAD" then
					updateUIRunsStatus("Waiting for floor assets...")
					local activeFloor = getActiveFloor(genMap, root)
					if activeFloor and currentRoom.Parent ~= activeFloor then
						state = "EVALUATE"
					end
				end

			elseif state == "HANDLE_FLOOR_3" then
				local doorsModel = currentRoom:FindFirstChild("Doors")
				local door = doorsModel and doorsModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState == "INIT" then
					updateUIRunsStatus("Approaching Boss Chamber...")
					if not door then
						if doorsModel and doorsModel:GetPivot() then
							preventFling(doorsModel:GetPivot() + Vector3.new(0, 5, 0))
						else
							preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0))
						end
						task.wait(1)
					else
						if (door.Position - root.Position).Magnitude > 15 then
							preventFling(door.CFrame * CFrame.new(0, 0, 10))
							task.wait(1)
						else
							roomState = "OPEN_DOOR"
						end
					end

				elseif roomState == "OPEN_DOOR" then
					updateUIRunsStatus("Opening Boss Chamber...")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0
						pcall(function() fireproximityprompt(prompt) end)
						task.wait(1)
					elseif prompt and not prompt.Enabled then
						spawnTimer = tick()
						roomState = "WAIT_SPAWN"
					end
				
				elseif roomState == "WAIT_SPAWN" then
					updateUIRunsStatus("Waiting for Elven Sorceress...")
					if tick() - spawnTimer >= 5 then
						roomState = "COMBAT"
					end

				elseif roomState == "COMBAT" then
					updateUIRunsStatus("Defeating Elven Sorceress...")
					local elvenAlive = false
					local mobsFolder = workspace:FindFirstChild("Mobs")
					
					if mobsFolder then
						for _, mob in pairs(mobsFolder:GetChildren()) do
							if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then
								elvenAlive = true
								break
							end
						end
					end

					if not elvenAlive then
						task.wait(1)
						local confirmAlive = false
						if mobsFolder then
							for _, mob in pairs(mobsFolder:GetChildren()) do
								if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then
									confirmAlive = true
									break
								end
							end
						end

						if not confirmAlive then
							_G.SlingshotElven = false
							preventFling(currentRoom:GetPivot() + Vector3.new(0,3,0))
							state = "EVALUATE"
							continue
						end
					end

					if not _G.SlingshotElven then
						_G.SlingshotElven = true
						SlingshotKill("Elven Sorceress", "SlingshotElven")
					end
				end

			elseif state == "HANDLE_LOOTING" then
				local chestsFolder = workspace:FindFirstChild("Chests")
				local mobsFolder = workspace:FindFirstChild("Mobs")
				
				local targetChest = nil
				local shortestDist = math.huge
				
				if chestsFolder then
					for _, chest in pairs(chestsFolder:GetChildren()) do
						if chest.Name == "TreasureChest" then
							local prompt = chest:FindFirstChildWhichIsA("ProximityPrompt", true)
							if prompt and prompt.Enabled then
								local dist = (chest:GetPivot().Position - root.Position).Magnitude
								if dist < shortestDist then
									shortestDist = dist
									targetChest = chest
								end
							end
						end
					end
				end

				if targetChest then
					updateUIRunsStatus("Collecting treasure chests...")
					root.CFrame = targetChest:GetPivot()
					
					local center = targetChest:FindFirstChild("Center")
					if center then
						root.CFrame = center.CFrame
					end
					
					local prompt = targetChest:FindFirstChildWhichIsA("ProximityPrompt", true)
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0
						task.wait(0.05)
						pcall(function() fireproximityprompt(prompt) end)
						
						local timeout = tick()
						while prompt.Enabled and tick() - timeout < 1.5 do
							task.wait(0.1)
							pcall(function() fireproximityprompt(prompt) end)
						end
					end
					continue
				end

				local targetMimic = nil
				shortestDist = math.huge
				
				if mobsFolder then
					for _, mob in pairs(mobsFolder:GetChildren()) do
						if mob.Name == "DungeonMimicChest" and isEntityAlive(mob) then
							local dist = (mob:GetPivot().Position - root.Position).Magnitude
							if dist < shortestDist then
								shortestDist = dist
								targetMimic = mob
							end
						end
					end
				end

				if targetMimic then
					local openVal = targetMimic:FindFirstChild("OpenChest")
					
					if openVal and openVal.Value == false then
						updateUIRunsStatus("Triggering Mimic Chest...")
						root.CFrame = targetMimic:GetPivot()
						
						local bitePlayer = targetMimic:FindFirstChild("BitePlayer") or targetMimic:FindFirstChild("IdleCollision")
						if bitePlayer then
							root.CFrame = bitePlayer.CFrame
						end
						
						local prompt = targetMimic:FindFirstChildWhichIsA("ProximityPrompt", true)
						if prompt and prompt.Enabled then
							prompt.HoldDuration = 0
							task.wait(0.05)
							pcall(function() fireproximityprompt(prompt) end)
							
							local timeout = tick()
							while targetMimic.Parent and tick() - timeout < 2 do
								if openVal.Value == true then break end
								task.wait(0.1)
								pcall(function() fireproximityprompt(prompt) end)
							end
						end

					elseif openVal and openVal.Value == true then
						updateUIRunsStatus("Defeating Mimic Hostile...")
						if not _G.SlingshotMimic then
							_G.SlingshotMimic = true
							SlingshotKill("DungeonMimicChest", "SlingshotMimic")
						end
						
						while _G.AutoDungeon and _G.SlingshotMimic and isEntityAlive(targetMimic) do
							task.wait(0.1) 
						end
						
						_G.SlingshotMimic = false
						task.wait(0.1)
					end
					task.wait(0.1)
					continue
				end

				-- Loot and boss variables are completely cleared; initiate natural dungeon exit phase
				updateUIRunsStatus("Looting completed. Leaving dungeon...")
				_G.KingsRuinsCompletedCount = _G.KingsRuinsCompletedCount + 1
				saveConfig()
				sendWebhookNotification("Kings Ruins Dungeon Completed Successfully!")
				
				teleportToExitGate()
				task.wait(3.0) 

				_G.AutoDungeon = false
				toggleNoclip(false)
				resetWalkSpeed()
			end

			task.wait(0.1)
		end
		
		toggleNoclip(false)
		_G.SlingshotStatue = false
		_G.SlingshotElven = false
		_G.SlingshotMimic = false
		resetWalkSpeed()
	end)
end

local function configureCameraForKingsRuins(enable)
	if enable then
		originalCameraType = Camera.CameraType
		originalCameraCFrame = Camera.CFrame
		
		local targetTeleport = workspace:FindFirstChild("KingsRuinsTeleport")
		if targetTeleport then
			Camera.CameraType = Enum.CameraType.Scriptable
			Camera.CFrame = CFrame.lookAt(targetTeleport:GetPivot().Position + Vector3.new(0, 50, -70), targetTeleport:GetPivot().Position)
		end
	else
		Camera.CameraType = originalCameraType
		Camera.CFrame = originalCameraCFrame
		
		local char = Player.Character
		local humanoid = char and char:FindFirstChildWhichIsA("Humanoid")
		if humanoid then
			Camera.CameraSubject = humanoid
		end
	end
end

local function handleVisualRadius(enable)
	if enable then
		if visualRadiusSphere then visualRadiusSphere:Destroy() end
		
		local targetTeleport = workspace:FindFirstChild("KingsRuinsTeleport")
		if targetTeleport then
			visualRadiusSphere = Instance.new("Part")
			visualRadiusSphere.Name = "Luna_RadiusVisualizer"
			visualRadiusSphere.Shape = Enum.PartType.Ball
			visualRadiusSphere.Size = Vector3.new(120, 120, 120) 
			visualRadiusSphere.Color = Color3.fromRGB(0, 255, 128)
			visualRadiusSphere.Material = Enum.Material.ForceField
			visualRadiusSphere.Transparency = 0.6
			visualRadiusSphere.Anchored = true
			visualRadiusSphere.CanCollide = false
			visualRadiusSphere.Position = targetTeleport:GetPivot().Position
			visualRadiusSphere.Parent = workspace
		end
	else
		if visualRadiusSphere then
			visualRadiusSphere:Destroy()
			visualRadiusSphere = nil
		end
	end
end

local function executeKingsRuinsEntrance()
	local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
	if teleportModel then
		local triggerPart = teleportModel:FindFirstChild("TeleportToHalloween")
		if triggerPart and triggerPart:FindFirstChildWhichIsA("TouchTransmitter") then
			local char = Player.Character
			local hrp = char and char:FindFirstChild("HumanoidRootPart")
			if hrp then
				firetouchinterest(hrp, triggerPart, 0)
				task.wait(0.05)
				firetouchinterest(hrp, triggerPart, 1)
			end
		end
	end
end

local function executeKingsRuinsLeave()
	local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
	if teleportModel then
		local barrier = teleportModel:FindFirstChild("LeaveBarrier")
		if barrier then
			local hrp = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
			if hrp then
				for _, part in pairs(barrier:GetChildren()) do
					if part:IsA("BasePart") and part:FindFirstChildWhichIsA("TouchTransmitter") then
						firetouchinterest(hrp, part, 0)
						task.wait(0.05)
						firetouchinterest(hrp, part, 1)
						break
					end
				end
			end
		end
	end
end

local function getLobbyPlayers()
	local lobbyPlayers = {}
	local listModel = workspace:FindFirstChild("KingsRuinsTeleport") and workspace.KingsRuinsTeleport:FindFirstChild("PlayerList")
	if listModel then
		for _, child in pairs(listModel:GetChildren()) do
			local nameValObj = child:FindFirstChild("Name")
			local playerName = ""
			if nameValObj then
				if nameValObj:IsA("StringValue") or nameValObj:IsA("TextValue") then
					playerName = nameValObj.Value
				else
					playerName = nameValObj.Name
				end
			else
				playerName = child.Name
			end
			if playerName ~= "" and Players:FindFirstChild(playerName) then
				table.insert(lobbyPlayers, playerName)
			end
		end
	end
	return lobbyPlayers
end

local function runKingsRuinsObserver()
	task.spawn(function()
		while _G.AutoKingsRuins do
			pcall(function()
				if isInDungeon() then
					-- If active dungeon clear is underway, skip status updates
					if not _G.AutoDungeon then
						updateUIRunsStatus("Dungeon active")
					end
					task.wait(2)
					return
				end

				local seasonStatus, remainingTime = getSeasonTimeRemaining()
				if seasonStatus == "Inactive" then
					if remainingTime > 120 then
						-- Grace period to allow client game values to synchronize on transition boundaries
						updateUIRunsStatus("Season shifting? Verifying...")
						task.wait(10)
						
						-- Re-verify season state after synchronization period
						seasonStatus, remainingTime = getSeasonTimeRemaining()
						if seasonStatus == "Inactive" and remainingTime > 120 then
							updateUIRunsStatus("Wrong season | Server Hopping")
							Luna:Notification({Title = "Auto Kings Ruins", Content = "Incorrect season. Hopping to target server Profile...", Icon = "autorenew"})
							task.wait(2)
							executeProjectedServerHop()
							return
						end
					else
						updateUIRunsStatus("Season starts in " .. math.floor(remainingTime) .. "s")
						task.wait(1)
						return
					end
				end

				local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
				if not teleportModel then
					updateUIRunsStatus("Teleporter model missing")
					task.wait(1)
					return
				end

				local isAnyPlayerInRadius = false
				local isAnyPlayerCloseToCenter = false
				local centerPos = teleportModel:GetPivot().Position
				local checkRadius = 60

				for _, otherPlayer in pairs(Players:GetPlayers()) do
					if otherPlayer ~= Player and not isPlayerAuthorized(otherPlayer.Name) and otherPlayer.Character and otherPlayer.Character:FindFirstChild("HumanoidRootPart") then
						local otherHRP = otherPlayer.Character.HumanoidRootPart
						local distance = (otherHRP.Position - centerPos).Magnitude
						if distance <= checkRadius then
							isAnyPlayerInRadius = true
							if distance <= 20 then
								isAnyPlayerCloseToCenter = true
							end
						end
					end
				end

				local lobbyPlayers = getLobbyPlayers()
				local amInLobby = table.find(lobbyPlayers, Player.Name) ~= nil
				local areOthersInLobby = false
				for _, name in ipairs(lobbyPlayers) do
					if name ~= Player.Name and not isPlayerAuthorized(name) then
						areOthersInLobby = true
						break
					end
				end

				local startsInText = ""
				local startsInObj = teleportModel:FindFirstChild("RaidPlayers") 
					and teleportModel.RaidPlayers:FindFirstChild("SurfaceGui") 
					and teleportModel.RaidPlayers.SurfaceGui:FindFirstChild("StartsIn")
				if startsInObj then
					startsInText = startsInObj.Text
				end

				local parsedSeconds = tonumber(string.match(startsInText, "%d+")) or 999
				local lobbyTimerActive = startsInText ~= ""

				if amInLobby then
					if areOthersInLobby then
						updateUIRunsStatus("Others inside lobby | Evacuating")
						executeKingsRuinsLeave()
						Luna:Notification({Title = "Auto Kings Ruins", Content = "Other players inside lobby. Evacuating...", Icon = "warning"})
						task.wait(2)
						return
					end

					if isAnyPlayerInRadius then
						if isAnyPlayerCloseToCenter then
							if not lobbyTimerActive or parsedSeconds > 3 then
								updateUIRunsStatus("Player near teleporter | Evacuating")
								executeKingsRuinsLeave()
								Luna:Notification({Title = "Auto Kings Ruins", Content = "Player detected near entrance. Evacuating...", Icon = "warning"})
								task.wait(2)
								return
							end
						end
					end
					
					updateUIRunsStatus("Waiting for start (" .. startsInText .. ")")
				else
					if isAnyPlayerInRadius then
						updateUIRunsStatus("Area blocked by players | Waiting patiently")
					else
						if startsInText == "TELEPORTING PARTY" then
							updateUIRunsStatus("Waiting for portal status to clear")
						elseif startsInText == "" then
							updateUIRunsStatus("Entering Lobby")
							executeKingsRuinsEntrance()
						else
							updateUIRunsStatus("Entering Lobby (Timer Active)")
							executeKingsRuinsEntrance()
						end
					end
				end
			end)
			task.wait(0.2)
		end
	end)
end

local currentMarkerList = {"None"} 

local function refreshMarkersList()
	local projFolder = workspace:FindFirstChild("Projectiles")
	local newList = {}
	local trackedNames = {}
	if projFolder then
		for _, proj in pairs(projFolder:GetChildren()) do
			if string.find(proj.Name, "Marker") and not trackedNames[proj.Name] then
				local markerPart = proj:FindFirstChild("Marker")
				if markerPart and markerPart:IsA("BasePart") then
					local bg = markerPart:FindFirstChildOfClass("BillboardGui") or markerPart:FindFirstChild("BillboardGui")
					if bg and bg.Enabled then
						trackedNames[proj.Name] = true
						table.insert(newList, proj.Name)
					end
				end
			end
		end
	end
	
	if #newList == 0 then 
		table.insert(newList, "None") 
	else
		table.sort(newList)
	end
	
	currentMarkerList = newList
	
	if not table.find(newList, _G.SelectedMarker) or _G.SelectedMarker == "None" then
		_G.SelectedMarker = newList[1]
	end

	if markerDropdown then
		markerDropdown:Set({Options = newList, CurrentOption = {_G.SelectedMarker}})
	end
end

local bypassWarning = false
local lastClickTime = 0

local function isPlayerNearDepartureOrArrival(targetPos, radius)
    for _, otherPlayer in pairs(Players:GetPlayers()) do
        if otherPlayer ~= Player and not isPlayerAuthorized(otherPlayer.Name) then
            local otherChar = otherPlayer.Character
            local otherRoot = otherChar and otherChar:FindFirstChild("HumanoidRootPart")
            if otherRoot then
                local otherPos = otherRoot.Position
                local distToArrival = (otherPos - targetPos).Magnitude
                
                -- Evaluates strictly whether unauthorized players are close to the target teleporter platform
                if distToArrival <= radius then
                    return true
                end
            end
        end
    end
    return false
end

    Name = "Teleport to KingsRuins",
    Description = "Tween 400 studs high and fall naturally onto Kings Ruins.",

        if not targetModel then
            Luna:Notification({
                Title = "Kings Ruins", 
                Content = "KingsRuinsTeleport target not found in workspace.", 
                Icon = "warning"
            })
            return
        end
        
        local char = Player.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then return end
        
        local targetPos = targetModel:GetPivot().Position
        local now = tick()
        
        local playerNear = isPlayerNearDepartureOrArrival(targetPos, 250)
        
        if playerNear then
            if not bypassWarning or (now - lastClickTime > 10) then
                bypassWarning = true
                lastClickTime = now
                Luna:Notification({
                    Title = "Safety Alert",
                    Content = "Another player is nearby! Click again within 10 seconds to force the tween.",
                    Icon = "warning"
                })
                return
            end
        end
        
        bypassWarning = false
        
        task.spawn(function()
            local noclipConn = RunService.Stepped:Connect(function()
                if char then
                    for _, p in pairs(char:GetDescendants()) do
                        if p:IsA("BasePart") then
                            p.CanCollide = false
                        end
                    end
                end
            end)
            
            local startPos = hrp.Position
            local peakY = math.max(startPos.Y, targetPos.Y) + 400
            
            local upCFrame = CFrame.new(startPos.X, peakY, startPos.Z)
            local distUp = math.abs(peakY - startPos.Y)
            local timeUp = distUp / 200 
            
            local upTween = TweenService:Create(hrp, TweenInfo.new(timeUp, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {CFrame = upCFrame})
            upTween:Play()
            upTween.Completed:Wait()
            
            local targetHighCFrame = CFrame.new(targetPos.X, peakY, targetPos.Z)
            local distHorizontal = (Vector3.new(targetPos.X, 0, targetPos.Z) - Vector3.new(startPos.X, 0, startPos.Z)).Magnitude
            local timeHorizontal = distHorizontal / 300 
            
            local horizTween = TweenService:Create(hrp, TweenInfo.new(timeHorizontal, Enum.EasingStyle.Linear), {CFrame = targetHighCFrame})
            horizTween:Play()
            horizTween.Completed:Wait()
            
            noclipConn:Disconnect()
            stopMomentum(hrp)
            
            Luna:Notification({
                Title = "Teleport",
                Content = "Arrived above Kings Ruins. Drop sequence active.",
                Icon = "check_circle_outline"
            })
        end)
    end

local secSeaFarms = tabSea:CreateSection("Sea Farming Automation")

local savedSeaPosition = nil

local function collectFinger(model)
	local character = Player.Character
	if not character or not character:FindFirstChild("HumanoidRootPart") then return end
	
	local hrp = character.HumanoidRootPart
	local touchedPart = model:FindFirstChild("Touched")
	
	if touchedPart then
		firetouchinterest(hrp, touchedPart, 0)
		task.wait()
		firetouchinterest(hrp, touchedPart, 1)
	else
		local originalCFrame = character:GetPivot() 
		local maxRetries = 3
		local currentRetry = 0
		
		while currentRetry < maxRetries and not touchedPart and _G.CursedFinger do
			character:PivotTo(model:GetPivot())
			task.wait(0.3) 
			
			touchedPart = model:FindFirstChild("Touched")
			if touchedPart then
				firetouchinterest(hrp, touchedPart, 0)
				task.wait()
				firetouchinterest(hrp, touchedPart, 1)
				break
			end
			
			currentRetry = currentRetry + 1
		end
		
		character:PivotTo(originalCFrame)
		task.wait(0.2) 
	end
end

secSeaFarms:CreateToggle({
	Name = "Auto Collect Cursed Fingers",
	Description = "Automatically teleport to and collect Cursed Fingers.",
	CurrentValue = false,
	Callback = function(val)
		_G.CursedFinger = val
		local character = Player.Character
		
		if val then
			if character then
				savedSeaPosition = character:GetPivot()
			end
			
			task.spawn(function()
				while _G.CursedFinger do
					local dropsContainer = workspace:FindFirstChild("Drops")
					if dropsContainer then
						local items = dropsContainer:GetChildren()
						for _, item in ipairs(items) do
							if not _G.CursedFinger then break end
							
							if item.Name == "CursedFinger" and item:IsA("Model") then
								collectFinger(item)
							end
						end
					end
					task.wait(1)
				end
			end)
		else
			if savedSeaPosition and character then
				character:PivotTo(savedSeaPosition)
				savedSeaPosition = nil
			end
		end
	end
})

local secWebhook = tabMisc:CreateSection("Webhook Integration")

-- Persistent textbox configuration utilizing the custom Input element added to the Luna section
local webhookInput = secWebhook:CreateInput({
	Name = "Discord Webhook",
	PlaceholderText = "Paste Discord Webhook URL here...",
	Enter = true,
	Callback = function(text)
		local sanitizedText = string.gsub(text, "%s+", "") -- Remove whitespaces
		if sanitizedText == "" or string.find(sanitizedText, "api/webhooks") then
			_G.DiscordWebhookURL = sanitizedText
			_G.DiscordMessageId = nil -- Reset in-memory ID on URL shifts
			saveSharedConfig() -- Writes to the SharedConfig file instantly
			sendWebhookNotification("ACTIVE", "Monitoring profile connected successfully.")
		else
			Luna:Notification({Title = "Webhook Config", Content = "Invalid Webhook URL pattern.", Icon = "warning"})
		end
	end
})

-- Update textbox visual state with the stored value on launch
webhookInput:Set({ CurrentValue = _G.DiscordWebhookURL })

if _G.TeleportConnection then
	_G.TeleportConnection:Disconnect()

end

if queue_on_teleport then
	_G.TeleportConnection = Players.LocalPlayer.OnTeleport:Connect(function(State)
		if not TeleportCheck then
			TeleportCheck = true
			
			queue_on_teleport([[
				if not game:IsLoaded() then
					game.Loaded:Wait()
				end
				task.wait(5)
				
				pcall(function()
					loadstring(game:HttpGet("https://raw.githubusercontent.com/NotShanks/test/refs/heads/master/uHub.txt"))()
				end)
			]])
		end
	end)
end

-- Post-Load Initialization Sequences
task.spawn(function()
	task.wait(1.5)
	if isInDungeon() then
		if _G.AutoKingsRuins then
			updateUIRunsStatus("Waiting for Map Load...")
			local genMap = workspace:FindFirstChild("GeneratedMap")
			local mapLoaded = genMap and genMap:WaitForChild("MapLoaded", 30)
			
			if mapLoaded then
				while mapLoaded.Value ~= true do
					task.wait(0.5)
				end
			end
			
			updateUIRunsStatus("Map verified. Loading assets (3s delay)...")
			task.wait(3)
			
			_G.AutoDungeon = true
			Luna:Notification({Title = "Auto Dungeon", Content = "Entering Kings Ruins. Initiating Dungeon automation...", Icon = "uautodungeon"})
			StartMasterAutoDungeon()
		end
	else
		if _G.AutoKingsRuins then
			configureCameraForKingsRuins(true)
			handleVisualRadius(true)
			runKingsRuinsObserver()
		end
	end
end)
