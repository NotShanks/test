local REAL_SECONDS_PER_DAY = 600
local IN_GAME_MINUTES_PER_DAY = 1440
local REAL_SECONDS_PER_GAME_MIN = REAL_SECONDS_PER_DAY / IN_GAME_MINUTES_PER_DAY 

local function getSeasonTimerText()
	local effectsFolder = workspace:FindFirstChild("Effects")
	local dayValueObj = effectsFolder and effectsFolder:FindFirstChild("Day")
	local ui = Player:FindFirstChild("PlayerGui")
	local timeTextObj = ui and ui:FindFirstChild("Moves") and ui.Moves:FindFirstChild("Bars") and ui.Moves.Bars:FindFirstChild("Time")

	if not dayValueObj or not timeTextObj then return "Error loading data" end
	local currentDay = dayValueObj.Value
	local remainder = currentDay % 8
	if remainder == 0 then remainder = 8 end

	local actionText = ""
	local daysDifference = 0
	if remainder == 3 or remainder == 4 then actionText = "Fall ends"; daysDifference = 5 - remainder
	elseif remainder == 7 or remainder == 8 then actionText = "Spring ends"; daysDifference = 9 - remainder
	elseif remainder == 1 or remainder == 2 then actionText = "Fall starts"; daysDifference = 3 - remainder 
	elseif remainder == 5 or remainder == 6 then actionText = "Spring starts"; daysDifference = 7 - remainder end

	local hoursStr, minutesStr = string.match(timeTextObj.Text, "(%d+):(%d+)")
	if not hoursStr or not minutesStr then return "Time parsing error" end
	
	local currentTotalGameMins = (tonumber(hoursStr) * 60) + tonumber(minutesStr)
	local gameMinsRemainingToday = IN_GAME_MINUTES_PER_DAY - currentTotalGameMins
	local totalGameMinsLeft = gameMinsRemainingToday + ((daysDifference - 1) * IN_GAME_MINUTES_PER_DAY)
	local realSecondsLeft = totalGameMinsLeft * REAL_SECONDS_PER_GAME_MIN

	return string.format("%s in %dm %ds", actionText, math.floor(realSecondsLeft / 60), math.floor(realSecondsLeft % 60))
end

local function getSeasonTimeRemaining()
	local effectsFolder = workspace:FindFirstChild("Effects")
	local dayValueObj = effectsFolder and effectsFolder:FindFirstChild("Day")
	local ui = Player:FindFirstChild("PlayerGui")
	local timeTextObj = ui and ui:FindFirstChild("Moves") and ui.Moves:FindFirstChild("Bars") and ui.Moves.Bars:FindFirstChild("Time")

	if not dayValueObj or not timeTextObj then return "Unknown", 9999 end
	local currentDay = dayValueObj.Value
	local remainder = currentDay % 8
	if remainder == 0 then remainder = 8 end

	local daysDifference = 0
	local isFallOrSpring = false
	if remainder == 3 or remainder == 4 then isFallOrSpring = true; daysDifference = 5 - remainder
	elseif remainder == 7 or remainder == 8 then isFallOrSpring = true; daysDifference = 9 - remainder
	elseif remainder == 1 or remainder == 2 then daysDifference = 3 - remainder 
	elseif remainder == 5 or remainder == 6 then daysDifference = 7 - remainder end

	local hoursStr, minutesStr = string.match(timeTextObj.Text, "(%d+):(%d+)")
	if not hoursStr or not minutesStr then return "Unknown", 9999 end

	local currentTotalGameMins = (tonumber(hoursStr) * 60) + tonumber(minutesStr)
	local gameMinsRemainingToday = IN_GAME_MINUTES_PER_DAY - currentTotalGameMins
	local totalGameMinsLeft = gameMinsRemainingToday + ((daysDifference - 1) * IN_GAME_MINUTES_PER_DAY)
	local realSecondsLeft = totalGameMinsLeft * REAL_SECONDS_PER_GAME_MIN

	return isFallOrSpring and "Active" or "Inactive", realSecondsLeft
end

local function FormatNumber(n)
	if not n then return "0" end
	n = tostring(math.floor(n))
	local formatted, k = n, nil
	while true do  
		formatted, k = string.gsub(formatted, "^(-?%d+)(%d%d%d)", '%1,%2')
		if k==0 then break end
	end
	return formatted
end

			local elapsedSeconds = math.max(1, os.time() - _G.StartupTime)

			local goldPerMin = math.floor(goldGained / elapsedMinutes)

			-- Set high score threshold check only after sufficient smoothing bounds pass to filter outliers (2 mins real active duration):
			if elapsedMinutes >= 2.0 and goldPerMin > _G.HighestGPM then
			    _G.HighestGPM = goldPerMin

			saveConfig()

			local uptimeStr = string.format("%dh %dm %ds", hours, minutes, seconds)

			if statusState == "FAILSAFE" then

local function updateCurrentServerCache()
	if isInDungeon() or game.JobId == "" then return end
	local effects = workspace:FindFirstChild("Effects")
	local dayObj = effects and effects:FindFirstChild("Day")
	local ui = Player:FindFirstChild("PlayerGui")
	local timeTextObj = ui and ui:FindFirstChild("Moves") and ui.Moves:FindFirstChild("Bars") and ui.Moves.Bars:FindFirstChild("Time")
	
	if dayObj and timeTextObj then
		local currentDay = dayObj.Value
		local hoursStr, minutesStr = string.match(timeTextObj.Text, "(%d+):(%d+)")
		if hoursStr and minutesStr then
			local gameMins = (tonumber(hoursStr) * 60) + tonumber(minutesStr)
			local cache = {}
			ensureConfigFolder()
			if isfile and isfile(cacheFileName) then pcall(function() cache = HttpService:JSONDecode(readfile(cacheFileName)) end) end
			cache[game.JobId] = { day = currentDay, gameMins = gameMins, realTime = os.time() }
			
			local now = os.time()
			for job, data in pairs(cache) do
				if now - data.realTime > 43200 then cache[job] = nil end
			end
			pcall(function() ensureConfigFolder() writefile(cacheFileName, HttpService:JSONEncode(cache)) end)
		end
	end
end

task.spawn(function()
	while task.wait(10) do pcall(updateCurrentServerCache) end
end)

local function getTargetServerFromCache()
	if not isfile or not isfile(cacheFileName) then return nil end
	
	local cache = {}
	pcall(function() cache = HttpService:JSONDecode(readfile(cacheFileName)) end)
	
	local now = os.time()
	local bestJobId, bestTimeRemaining = nil, 9999
	
	for jobId, data in pairs(cache) do
		if jobId ~= game.JobId and not _G.FailedCachedJobs[jobId] then
			local elapsedGameMins = ((now - data.realTime) / REAL_SECONDS_PER_DAY) * IN_GAME_MINUTES_PER_DAY
			local projectedTotalMins = data.gameMins + elapsedGameMins
			local totalGameDaysElapsed = math.floor(projectedTotalMins / IN_GAME_MINUTES_PER_DAY)
			local finalProjectedDay = data.day + totalGameDaysElapsed
			local currentMinsInDay = projectedTotalMins % IN_GAME_MINUTES_PER_DAY
			local remainder = finalProjectedDay % 8
			if remainder == 0 then remainder = 8 end
			
			local isFallOrSpring = (remainder == 3 or remainder == 4 or remainder == 7 or remainder == 8)
			local daysDifference = 0
			if remainder == 1 or remainder == 2 then daysDifference = 3 - remainder
			elseif remainder == 5 or remainder == 6 then daysDifference = 7 - remainder
			elseif remainder == 3 or remainder == 4 then daysDifference = 5 - remainder
			elseif remainder == 7 or remainder == 8 then daysDifference = 9 - remainder end
			
			local realSecondsLeft = (IN_GAME_MINUTES_PER_DAY - currentMinsInDay + ((daysDifference - 1) * IN_GAME_MINUTES_PER_DAY)) * REAL_SECONDS_PER_GAME_MIN
			if isFallOrSpring or realSecondsLeft <= 120 then
				if realSecondsLeft < bestTimeRemaining then
					bestTimeRemaining, bestJobId = realSecondsLeft, jobId
				end
			end
		end
	end
	return bestJobId
end

doServerHop = function()
	local PlaceId = game.PlaceId
	local JobId = game.JobId
	if isHoppingCurrently then return end
	isHoppingCurrently = true
	updateUIRunsStatus("Searching for reliable lobbies...")
	
	task.spawn(function()
		local success, result = pcall(function()
			local req = game:HttpGet("https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Asc&limit=100&excludeFullGames=true")
			local body = HttpService:JSONDecode(req)
			local validServers = {}
			if body and body.data then
				for _, v in pairs(body.data) do
					if type(v) == "table" and tonumber(v.playing) and tonumber(v.maxPlayers) then
						local players, max = tonumber(v.playing), tonumber(v.maxPlayers)
						if players >= 2 and players < (max - 2) and v.id ~= JobId then table.insert(validServers, v.id) end
					end
				end
			end

			if #validServers > 0 then
				local selectedInstance = validServers[math.random(1, #validServers)]
				_G.LastTargetedJobId = selectedInstance
				TeleportService:TeleportToPlaceInstance(PlaceId, selectedInstance, Player)
			else
				updateUIRunsStatus("Server map locked. Standard rejoining...")
				TeleportService:Teleport(PlaceId, Player)
			end
		end)
		
		if not success then TeleportService:Teleport(PlaceId, Player) end
		task.wait(12)
		isHoppingCurrently = false
	end)
end

local function executeProjectedServerHop()
	if isHoppingCurrently then return end 
	
	local cachedJobId = getTargetServerFromCache()
	if cachedJobId and _G.ConsecutiveHopAttempts < 4 then
		_G.ConsecutiveHopAttempts = _G.ConsecutiveHopAttempts + 1

		task.spawn(function()
			pcall(function() TeleportService:TeleportToPlaceInstance(game.PlaceId, cachedJobId, Player) end)
			task.wait(12)
			isHoppingCurrently = false
		end)
	else
		if _G.ConsecutiveHopAttempts >= 4 then
			updateUIRunsStatus("Target failed/full. Hopping public!")
			if cachedJobId then _G.FailedCachedJobs[cachedJobId] = true end
			task.spawn(function() Luna:Notification({Title="Routing Overload", Content="Target Profile full or inaccessible! Falling back.", Icon="warning"}) end)

		_G.ConsecutiveHopAttempts = 0

		doServerHop()
	end
end

_G.CancelTween = false
local function doSmartTween(targetPos)
	if _G.EnableTweeningSafety == false then return end
	_G.CancelTween = false
	local char = Player.Character
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	local humanoid = char and char:FindFirstChildWhichIsA("Humanoid")
	if not char or not hrp or not humanoid then return end

	local tpSpeed = 200 
	humanoid.PlatformStand = true 
	
	local bv = Instance.new("BodyVelocity")
	bv.Velocity = Vector3.new(0, 0, 0); bv.MaxForce = Vector3.new(9e9, 9e9, 9e9); bv.Parent = hrp
	local noclipConn = RunService.Stepped:Connect(function() for _, v in pairs(char:GetDescendants()) do if v:IsA("BasePart") and v.CanCollide then v.CanCollide = false end end end)

	local function tpWalkTo(targetVec)
		local reached = false
		local tpConnection
		tpConnection = RunService.Heartbeat:Connect(function(delta)
			if _G.CancelTween or not char or not hrp or not hrp.Parent then
				reached = true; if tpConnection then tpConnection:Disconnect() end; return
			end
			local currentPos = hrp.Position
			local dist = (targetVec - currentPos).Magnitude
			if dist < 5 then reached = true; if tpConnection then tpConnection:Disconnect() end; return end
			local direction = (targetVec - currentPos).Unit
			local moveAmount = direction * tpSpeed * delta * 10
			if moveAmount.Magnitude > dist then char:TranslateBy(direction * dist) reached = true if tpConnection then tpConnection:Disconnect() end else char:TranslateBy(moveAmount) end
			hrp.Velocity = Vector3.zero; hrp.AssemblyLinearVelocity = Vector3.zero
		end)
		while not reached do task.wait() end
	end

	if not _G.CancelTween then tpWalkTo(hrp.Position + Vector3.new(0, 300, 0)) end
	if not _G.CancelTween then tpWalkTo(Vector3.new(targetPos.X, hrp.Position.Y + 300, targetPos.Z)) end
	if not _G.CancelTween then tpWalkTo(targetPos + Vector3.new(0, 3, 0)) end
	if bv then bv:Destroy() end
	if noclipConn then noclipConn:Disconnect() end
	if humanoid then humanoid.PlatformStand = false end
	if hrp then stopMomentum(hrp) end
end

local activeNoclip, activeFloat, currentTweenObj
local function CancelSmoothTween()
	if currentTweenObj then currentTweenObj:Cancel() currentTweenObj = nil end
	if activeNoclip then activeNoclip:Disconnect() activeNoclip = nil end
	if activeFloat then activeFloat:Destroy() activeFloat = nil end
	local char = Player.Character
	if char and char:FindFirstChild("HumanoidRootPart") then stopMomentum(char.HumanoidRootPart) end
end

local function SmoothTween(targetCFrame, speed, conditionCheck)
	local char = Player.Character; if not char then return false end
	local hrp = char:FindFirstChild("HumanoidRootPart"); if not hrp then return false end
	CancelSmoothTween()
	local dist = (hrp.Position - targetCFrame.Position).Magnitude
	local tweenTime = dist / speed
	
	activeFloat = Instance.new("BodyVelocity")
	activeFloat.Name = "SmoothFloat"
	activeFloat.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
	activeFloat.Velocity = Vector3.zero
	activeFloat.Parent = hrp
	activeNoclip = RunService.Stepped:Connect(function() for _, v in pairs(char:GetDescendants()) do if v:IsA("BasePart") then v.CanCollide = false end end end)
	
	currentTweenObj = TweenService:Create(hrp, TweenInfo.new(tweenTime, Enum.EasingStyle.Linear), {CFrame = targetCFrame}); currentTweenObj:Play()
	local startTime = tick()
	while tick() - startTime < tweenTime do
		if conditionCheck and not conditionCheck() then CancelSmoothTween() return false end
		task.wait(0.05)
	end
	CancelSmoothTween()
	return true
end

local function getGates()
	local list = {}; local refs = {}
	local gatesFolder = workspace:FindFirstChild("Gates")
	if gatesFolder then
		for _, gateModel in pairs(gatesFolder:GetChildren()) do
			if gateModel:IsA("Model") then
				local levelObj, rankObj, timeSpawnedObj = gateModel:FindFirstChild("Level", true), gateModel:FindFirstChild("Rank", true), gateModel:FindFirstChild("TimeSpawned", true)
				local timeLeftStr = ""
				if timeSpawnedObj and tonumber(timeSpawnedObj.Value) then
					local remaining = 550 - (os.time() - tonumber(timeSpawnedObj.Value))
					if remaining > 0 then timeLeftStr = string.format(" | %02d:%02d", math.floor(remaining/60), remaining%60)
					else timeLeftStr = " | Expired" end
				end
				local name = string.format("%s [Lvl %s | Rank %s]%s", gateModel.Name, levelObj and tostring(levelObj.Value) or "?", rankObj and tostring(rankObj.Value) or "?", timeLeftStr)
				table.insert(list, name); refs[name] = gateModel
			end
		end
	end
	if #list == 0 then table.insert(list, "No Gates Found") end
	return list, refs
end

			local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")

			if workspace:FindFirstChild("Mobs") and myRoot then
				for _, v in pairs(workspace.Mobs:GetChildren()) do
					if v.Name ~= "Regulus" and v:FindFirstChild("HumanoidRootPart") and isEntityAlive(v) then
						local dist = (v.HumanoidRootPart.Position - myRoot.Position).Magnitude
						if dist < shortestDist then shortestDist = dist targetMob = v end

			end

				local camConn = RunService.RenderStepped:Connect(function()
					if targetMob and targetMob:FindFirstChild("HumanoidRootPart") then cam.CFrame = CFrame.lookAt(targetMob.HumanoidRootPart.Position + Vector3.new(0, 30, 40), targetMob.HumanoidRootPart.Position) end
				end)

				local lastHRFire = 0
				local entityFocusTimer = tick() -- Time loop limit to force escape loops mapping
				
				while _G[toggleKey] and isEntityAlive(targetMob) and myChar and myChar:FindFirstChild("HumanoidRootPart") do
					if tick() - lastHRFire >= 1 then triggerHeavenlyRestriction() lastHRFire = tick() end
					if isInDungeon() and tick() - entityFocusTimer > 120 then 
					    triggerDungeonSuicide("Stuck hitting single generic map target inside dungeon >2 Mins.")

					    break
					end

			end

		end

	end)

local function getFurthestPart(roomModel, rootPosition)
	local furthestPart, maxDist = nil, 0
	for _, obj in pairs(roomModel:GetDescendants()) do if obj:IsA("BasePart") then local dist = (obj.Position - rootPosition).Magnitude if dist > maxDist and dist < 200 then maxDist = dist; furthestPart = obj end end end
	return furthestPart
end

local function getActiveFloor(genMap, root)
	local closestFloor, minDist = nil, math.huge
	for i = 1, 3 do local f = genMap:FindFirstChild("Floor"..i); if f then for _, room in pairs(f:GetChildren()) do if room:IsA("Model") and room:GetPivot() then local d = (room:GetPivot().Position - root.Position).Magnitude if d < minDist then minDist, closestFloor = d, f end end end end end
	return closestFloor
end

local function getIncompleteEventRoom(floorModel)
	if not floorModel then return nil end
	for _, room in pairs(floorModel:GetChildren()) do if room.Name == "EventRoom" then local comp = room:FindFirstChild("Complete"); if comp and comp.Value == false then return room end end end
	return nil
end

local function isExitRoomReady(floorModel)
	if not floorModel then return false end; local exitRoom = floorModel:FindFirstChild("ExitRoom"); if not exitRoom then return false end
	local e, c = 0, 0
	for _, child in ipairs(exitRoom:GetChildren()) do if string.match(child.Name, "Event%d+") then e = e + 1; local comp = child:FindFirstChild("Complete"); if comp and comp.Value == true then c = c + 1 end end end
	return e > 0 and e == c
end

local function teleportToExitGate()
	local targetGate = nil
	for _, obj in pairs(workspace:GetDescendants()) do if obj.Name == "ExitGate" then targetGate = obj; break end end
	if targetGate then
		local pos = targetGate:IsA("Model") and targetGate:GetPivot().Position or targetGate.Position
		local hrp = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
		if hrp then hrp.CFrame = CFrame.new(pos) * (hrp.CFrame - hrp.CFrame.Position); Luna:Notification({Title = "Kings Ruins", Content = "Transitioning to ExitGate.", Icon = "check_circle_outline"}) end
	end
end

local function isDungeonClear()
	local m, c = workspace:FindFirstChild("Mobs"), workspace:FindFirstChild("Chests")
	local aMobs, aChests = 0, 0
	if m then for _, mob in pairs(m:GetChildren()) do if mob:IsA("Model") and isEntityAlive(mob) and mob.Name ~= "DungeonMimicChest" then aMobs = aMobs + 1 end end end
	if c then for _, chest in pairs(c:GetChildren()) do if chest.Name == "TreasureChest" or chest.Name == "SmallChest" then local prompt = chest:FindFirstChildWhichIsA("ProximityPrompt", true) if prompt and prompt.Enabled then aChests = aChests + 1 end end end end
	return (aMobs == 0 and aChests == 0)
end

local function startExitCheckWatcher()
	task.spawn(function()
		while _G.AutoDungeon do
			local genMap = workspace:FindFirstChild("GeneratedMap"); local char = Player.Character; local root = char and char:FindFirstChild("HumanoidRootPart")
			if genMap and root then
				local activeFloor = getActiveFloor(genMap, root)
				if activeFloor and activeFloor.Name == "Floor3" then
					local arena = activeFloor:FindFirstChild("KingsRuinsArena")
					if arena then
						local door = arena:FindFirstChild("Doors") and arena.Doors:FindFirstChild("DoorMain")
						local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)
						if prompt and not prompt.Enabled and isDungeonClear() then
							_G.KingsRuinsCompletedCount = _G.KingsRuinsCompletedCount + 1; saveConfig(); sendWebhookNotification("Kings Ruins Completed!")
							teleportToExitGate(); break
						end
					end
				end
			end
			task.wait(1)
		end
	end)
end

local function StartMasterAutoDungeon()
	task.spawn(function()
		local state = "EVALUATE"; local currentRoom, lastRoom, currentTween = nil, nil, nil
		local roomState = "INIT"; local lastTrapClear, spawnTimer = 0, 0

		startExitCheckWatcher()

		while _G.AutoDungeon do
			if tick() - lastTrapClear > 1 then
				lastTrapClear = tick()
				pcall(function() for _, obj in pairs(workspace.GeneratedMap:GetDescendants()) do if obj.Name == "TrapStorage" then obj:Destroy() end end end)
			end

			local char = Player.Character; local root = char and char:FindFirstChild("HumanoidRootPart")
			local genMap = workspace:FindFirstChild("GeneratedMap")
			if not genMap or not root then task.wait(0.5) continue end

			if state == "EVALUATE" then
				updateUIRunsStatus("Evaluating Active Floor...")
				local activeFloor = getActiveFloor(genMap, root)
				if activeFloor then
					if activeFloor.Name == "Floor3" then
						local arena = activeFloor:FindFirstChild("KingsRuinsArena")
						if arena then
							local bossAlive = false
							local mobsFolder = workspace:FindFirstChild("Mobs")
							if mobsFolder then for _, mob in pairs(mobsFolder:GetChildren()) do if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then bossAlive = true break end end end
							local door = arena:FindFirstChild("Doors") and arena.Doors:FindFirstChild("DoorMain")
							local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)
							
							if prompt and not prompt.Enabled and not bossAlive then currentRoom = arena; state = "HANDLE_LOOTING" else currentRoom = arena; state = "HANDLE_FLOOR_3" end
						end
					else
						local room = getIncompleteEventRoom(activeFloor)
						if room then currentRoom = room; state = "HANDLE_EVENT_ROOM" elseif isExitRoomReady(activeFloor) then currentRoom = activeFloor:FindFirstChild("ExitRoom"); state = "HANDLE_EXIT_ROOM" end
					end
				end
				if currentRoom ~= lastRoom then roomState = "INIT"; lastRoom = currentRoom end

			elseif state == "HANDLE_EVENT_ROOM" then
				local comp = currentRoom:FindFirstChild("Complete")
				if comp and comp.Value == true then
					_G.SlingshotStatue = false
					if currentTween then currentTween:Cancel(); currentTween = nil end
					preventFling(currentRoom:GetPivot() + Vector3.new(0,3,0)); state = "EVALUATE"; continue
				end

				local doorsModel = currentRoom:FindFirstChild("Doors")
				local door = doorsModel and doorsModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState ~= "INIT" and roomState ~= "OPEN_DOOR" and prompt and prompt.Enabled then
					if currentTween then currentTween:Cancel(); currentTween = nil end; _G.SlingshotStatue = false; roomState = "INIT"; continue
				end

				if roomState == "INIT" then
					updateUIRunsStatus("Moving to event room gate...")
					_G.DoorStuckTimer = tick() -- Suicide evaluation base timer starts here dynamically per unique interaction loop
					if not door then
						if doorsModel and doorsModel:GetPivot() then preventFling(doorsModel:GetPivot() + Vector3.new(0, 5, 0)) else preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0)) end; task.wait(1)
					else
						local doorCF = door.CFrame
						local localPos = doorCF:PointToObjectSpace(root.Position)
						local safePos

						if math.abs(localPos.Z) > math.abs(localPos.X) then safePos = doorCF * Vector3.new(0, 0, math.sign(localPos.Z) * 8)
						else safePos = doorCF * Vector3.new(math.sign(localPos.X) * 8, 0, 0) end

						preventFling(CFrame.lookAt(safePos, door.Position)); _G.DoorRetryDistance = 8; task.wait(0.5); roomState = "OPEN_DOOR"
					end

				elseif roomState == "OPEN_DOOR" then
					if _G.DoorStuckTimer and tick() - _G.DoorStuckTimer > 60 then
					    triggerDungeonSuicide("Blocked indefinitely at locked room door! [Stall Timeout -> 60s]") 
					    break
					end
				
					updateUIRunsStatus("Opening gate [Dist: " .. math.round(_G.DoorRetryDistance or 8) .. "]")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0; pcall(function() fireproximityprompt(prompt) end); task.wait(0.7)
						if prompt.Enabled then
							_G.DoorRetryDistance = math.max((_G.DoorRetryDistance or 8) - 1.5, 0.5)
							local doorCF = door.CFrame; local localPos = doorCF:PointToObjectSpace(root.Position); local closerPos
							if math.abs(localPos.Z) > math.abs(localPos.X) then closerPos = doorCF * Vector3.new(0, 0, math.sign(localPos.Z) * _G.DoorRetryDistance)
							else closerPos = doorCF * Vector3.new(math.sign(localPos.X) * _G.DoorRetryDistance, 0, 0) end
							preventFling(CFrame.lookAt(closerPos, door.Position))
						end
					elseif prompt and not prompt.Enabled then _G.DoorRetryDistance = nil; _G.DoorStuckTimer = nil; roomState = "WALK_IN" end

				elseif roomState == "WALK_IN" then
					updateUIRunsStatus("Advancing into room...")
					local closeSound = currentRoom:FindFirstChild("Doors") and currentRoom.Doors:FindFirstChild("WallBlock") and currentRoom.Doors.WallBlock:FindFirstChild("Close")
					if closeSound and closeSound.Playing then
						if currentTween then currentTween:Cancel(); currentTween = nil end
						toggleNoclip(false); preventFling(); roomState = "WAIT_CLOSE"
					else
						if not currentTween or currentTween.PlaybackState ~= Enum.PlaybackState.Playing then
							local furthestPart = getFurthestPart(currentRoom, root.Position)
							local targetPos = furthestPart and furthestPart.Position or currentRoom:GetPivot().Position
							targetPos = Vector3.new(targetPos.X, currentRoom:GetPivot().Position.Y + 3, targetPos.Z)
							toggleNoclip(true); currentTween = TweenService:Create(root, TweenInfo.new((root.Position - targetPos).Magnitude / 70, Enum.EasingStyle.Linear), {CFrame = CFrame.new(targetPos)}); currentTween:Play()
						end
					end

				elseif roomState == "WAIT_CLOSE" then
					updateUIRunsStatus("Waiting for entry gate to lock...")
					local closeSound = currentRoom:FindFirstChild("Doors") and currentRoom.Doors:FindFirstChild("WallBlock") and currentRoom.Doors.WallBlock:FindFirstChild("Close")
					if closeSound and not closeSound.Playing then roomState = "COMBAT" elseif not closeSound then roomState = "COMBAT" end

				elseif roomState == "COMBAT" then
					updateUIRunsStatus("Clearing event guardians...")
					if not _G.SlingshotStatue then _G.SlingshotStatue = true; SlingshotKill("Statue", "SlingshotStatue") end
				end

			elseif state == "HANDLE_EXIT_ROOM" then
				local gateModel = currentRoom:FindFirstChild("Gate")
				local door = gateModel and gateModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState == "INIT" then
					updateUIRunsStatus("Navigating to exit gate...")
					_G.DoorStuckTimer = tick() -- Arm fail safe
					if not door then
						if gateModel and gateModel:GetPivot() then preventFling(gateModel:GetPivot() + Vector3.new(0, 5, 0)) else preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0)) end; task.wait(1)
					else
						preventFling(door.CFrame * CFrame.new(0, 0, 8)); _G.DoorRetryDistance = 8; task.wait(1); roomState = "OPEN_GATE"
					end
				
				elseif roomState == "OPEN_GATE" then
				    if _G.DoorStuckTimer and tick() - _G.DoorStuckTimer > 60 then triggerDungeonSuicide("Failure mapping End Gate connection layout! [Stall Timeout -> 60s]") break end
				    
					updateUIRunsStatus("Opening floor exit portal [Dist: " .. math.round(_G.DoorRetryDistance or 8) .. "]")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0; pcall(function() fireproximityprompt(prompt) end); task.wait(0.7)
						if prompt.Enabled then _G.DoorRetryDistance = math.max((_G.DoorRetryDistance or 8) - 1.5, 0.5); preventFling(door.CFrame * CFrame.new(0, 0, _G.DoorRetryDistance)) end
					elseif prompt and not prompt.Enabled then _G.DoorRetryDistance = nil; _G.DoorStuckTimer = nil; roomState = "WALK_THROUGH" end

				elseif roomState == "WALK_THROUGH" then
					updateUIRunsStatus("Entering next floor portal...")
					if door then
						local nextStart = currentRoom:FindFirstChild("NextFloorStart")
						if nextStart then preventFling(nextStart.CFrame + Vector3.new(0, 5, 0)) else preventFling(door.CFrame * CFrame.new(0, 0, -25)) end
						task.wait(3.5); roomState = "WAIT_FLOOR_LOAD"
					end

				elseif roomState == "WAIT_FLOOR_LOAD" then
					updateUIRunsStatus("Waiting for floor assets...")
					local activeFloor = getActiveFloor(genMap, root)
					if activeFloor and currentRoom.Parent ~= activeFloor then state = "EVALUATE" end
				end

			elseif state == "HANDLE_FLOOR_3" then
				local doorsModel = currentRoom:FindFirstChild("Doors")
				local door = doorsModel and doorsModel:FindFirstChild("DoorMain")
				local prompt = door and door:FindFirstChildWhichIsA("ProximityPrompt", true)

				if roomState == "INIT" then
					updateUIRunsStatus("Approaching Boss Chamber...")
					_G.DoorStuckTimer = tick() -- Prime suicide exit block timer
					if not door then
						if doorsModel and doorsModel:GetPivot() then preventFling(doorsModel:GetPivot() + Vector3.new(0, 5, 0)) else preventFling(currentRoom:GetPivot() + Vector3.new(0, 5, 0)) end; task.wait(1)
					else preventFling(door.CFrame * CFrame.new(0, 0, 10)); _G.DoorRetryDistance = 10; task.wait(1); roomState = "OPEN_DOOR" end

				elseif roomState == "OPEN_DOOR" then
				    if _G.DoorStuckTimer and tick() - _G.DoorStuckTimer > 60 then triggerDungeonSuicide("Inable to process final raid chamber threshold limit [Stall Timeout -> 60s]") break end
				    
					updateUIRunsStatus("Opening Boss Chamber [Dist: " .. math.round(_G.DoorRetryDistance or 10) .. "]")
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0; pcall(function() fireproximityprompt(prompt) end); task.wait(0.7)
						if prompt.Enabled then _G.DoorRetryDistance = math.max((_G.DoorRetryDistance or 10) - 1.5, 0.5); preventFling(door.CFrame * CFrame.new(0, 0, _G.DoorRetryDistance)) end
					elseif prompt and not prompt.Enabled then _G.DoorRetryDistance = nil; _G.DoorStuckTimer = nil; spawnTimer = tick(); roomState = "WAIT_SPAWN" end
				
				elseif roomState == "WAIT_SPAWN" then
					updateUIRunsStatus("Waiting for Elven Sorceress...")
					if tick() - spawnTimer >= 5 then roomState = "COMBAT" end

				elseif roomState == "COMBAT" then
					updateUIRunsStatus("Defeating Elven Sorceress...")
					local elvenAlive, mobsFolder = false, workspace:FindFirstChild("Mobs")
					if mobsFolder then for _, mob in pairs(mobsFolder:GetChildren()) do if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then elvenAlive = true break end end end

					if not elvenAlive then
						task.wait(1); local confirmAlive = false
						if mobsFolder then for _, mob in pairs(mobsFolder:GetChildren()) do if mob.Name == "Elven Sorceress" and isEntityAlive(mob) then confirmAlive = true break end end end
						if not confirmAlive then _G.SlingshotElven = false; preventFling(currentRoom:GetPivot() + Vector3.new(0,3,0)); state = "EVALUATE"; continue end
					end

					if not _G.SlingshotElven then _G.SlingshotElven = true; SlingshotKill("Elven Sorceress", "SlingshotElven") end
				end

			elseif state == "HANDLE_LOOTING" then
				local chestsFolder, mobsFolder = workspace:FindFirstChild("Chests"), workspace:FindFirstChild("Mobs")
				local targetChest, shortestDist = nil, math.huge
				
				if chestsFolder then
					for _, chest in pairs(chestsFolder:GetChildren()) do
						if chest.Name == "TreasureChest" then
							local prompt = chest:FindFirstChildWhichIsA("ProximityPrompt", true)
							if prompt and prompt.Enabled then local dist = (chest:GetPivot().Position - root.Position).Magnitude if dist < shortestDist then shortestDist, targetChest = dist, chest end end
						end
					end
				end

				if targetChest then
					updateUIRunsStatus("Collecting treasure chests...")
					local center = targetChest:FindFirstChild("Center")
					root.CFrame = center and center.CFrame or targetChest:GetPivot()
					
					local prompt = targetChest:FindFirstChildWhichIsA("ProximityPrompt", true)
					if prompt and prompt.Enabled then
						prompt.HoldDuration = 0; task.wait(0.05); pcall(function() fireproximityprompt(prompt) end)
						local timeout = tick()
						while prompt.Enabled and tick() - timeout < 1.5 do task.wait(0.1); pcall(function() fireproximityprompt(prompt) end) end
					end
					continue
				end

				local targetMimic = nil; shortestDist = math.huge
				if mobsFolder then
					for _, mob in pairs(mobsFolder:GetChildren()) do
						if mob.Name == "DungeonMimicChest" and isEntityAlive(mob) then
							local dist = (mob:GetPivot().Position - root.Position).Magnitude
							if dist < shortestDist then shortestDist, targetMimic = dist, mob end
						end
					end
				end

				if targetMimic then
					local openVal = targetMimic:FindFirstChild("OpenChest")
					if openVal and openVal.Value == false then
						updateUIRunsStatus("Triggering Mimic Chest...")
						local bitePlayer = targetMimic:FindFirstChild("BitePlayer") or targetMimic:FindFirstChild("IdleCollision")
						root.CFrame = bitePlayer and bitePlayer.CFrame or targetMimic:GetPivot()
						
						local prompt = targetMimic:FindFirstChildWhichIsA("ProximityPrompt", true)
						if prompt and prompt.Enabled then
							prompt.HoldDuration = 0; task.wait(0.05); pcall(function() fireproximityprompt(prompt) end)
							local timeout = tick()
							while targetMimic.Parent and tick() - timeout < 2 do
								if openVal.Value == true then break end
								task.wait(0.1); pcall(function() fireproximityprompt(prompt) end)
							end
						end
					elseif openVal and openVal.Value == true then
						updateUIRunsStatus("Defeating Mimic Hostile...")
						if not _G.SlingshotMimic then _G.SlingshotMimic = true; SlingshotKill("DungeonMimicChest", "SlingshotMimic") end
						while _G.AutoDungeon and _G.SlingshotMimic and isEntityAlive(targetMimic) do task.wait(0.1) end
						_G.SlingshotMimic = false; task.wait(0.1)
					end
					task.wait(0.1); continue
				end

				updateUIRunsStatus("Looting completed. Leaving dungeon...")
				_G.KingsRuinsCompletedCount = _G.KingsRuinsCompletedCount + 1; saveConfig()
				sendWebhookNotification("Kings Ruins Dungeon Completed Successfully!")
				teleportToExitGate(); task.wait(3.0) 

				_G.AutoDungeon = false; toggleNoclip(false); resetWalkSpeed()
			end

			task.wait(0.1)
		end
		
		toggleNoclip(false); _G.SlingshotStatue = false; _G.SlingshotElven = false; _G.SlingshotMimic = false; resetWalkSpeed()
	end)
end

local function executeKingsRuinsEntrance()
	local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
	if teleportModel then
		local triggerPart = teleportModel:FindFirstChild("TeleportToHalloween")
		if triggerPart and triggerPart:FindFirstChildWhichIsA("TouchTransmitter") then
			local hrp = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
			if hrp then firetouchinterest(hrp, triggerPart, 0) task.wait(0.05) firetouchinterest(hrp, triggerPart, 1) end
		end
	end
end

local function executeKingsRuinsLeave()
	local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
	if teleportModel then
		local barrier = teleportModel:FindFirstChild("LeaveBarrier")
		local hrp = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
		if barrier and hrp then
			for _, part in pairs(barrier:GetChildren()) do
				if part:IsA("BasePart") and part:FindFirstChildWhichIsA("TouchTransmitter") then firetouchinterest(hrp, part, 0) task.wait(0.05) firetouchinterest(hrp, part, 1) break end
			end
		end
	end
end

local function getLobbyPlayers()
	local lobbyPlayers = {}; local listModel = workspace:FindFirstChild("KingsRuinsTeleport") and workspace.KingsRuinsTeleport:FindFirstChild("PlayerList")
	if listModel then
		for _, child in pairs(listModel:GetChildren()) do
			local nameValObj, playerName = child:FindFirstChild("Name"), ""
			if nameValObj then if nameValObj:IsA("StringValue") or nameValObj:IsA("TextValue") then playerName = nameValObj.Value else playerName = nameValObj.Name end else playerName = child.Name end
			if playerName ~= "" and Players:FindFirstChild(playerName) then table.insert(lobbyPlayers, playerName) end
		end
	end
	return lobbyPlayers
end

local function runKingsRuinsObserver()
	task.spawn(function()
		while _G.AutoKingsRuins do
			pcall(function()
				if isInDungeon() then if not _G.AutoDungeon then updateUIRunsStatus("Dungeon active") end; task.wait(2) return end
				local seasonStatus, remainingTime = getSeasonTimeRemaining()
				if seasonStatus == "Inactive" then
					if remainingTime > 120 then
						updateUIRunsStatus("Season shifting? Verifying...")
						task.wait(10); seasonStatus, remainingTime = getSeasonTimeRemaining()
						if seasonStatus == "Inactive" and remainingTime > 120 then updateUIRunsStatus("Wrong season | Server Hopping"); task.wait(2); executeProjectedServerHop(); return end
					else updateUIRunsStatus("Season starts in " .. math.floor(remainingTime) .. "s"); task.wait(1); return end
				end

				local teleportModel = workspace:FindFirstChild("KingsRuinsTeleport")
				if not teleportModel then updateUIRunsStatus("Teleporter model missing"); task.wait(1); return end
				local centerPos = teleportModel:GetPivot().Position
				local lobbyPlayers = getLobbyPlayers()
				local amInLobby, areOthersInLobby = table.find(lobbyPlayers, Player.Name) ~= nil, false
				
				for _, name in ipairs(lobbyPlayers) do if name ~= Player.Name and not isPlayerAuthorized(name) then areOthersInLobby = true end end

				local playerInZone1, playerInZone2, playerInZone3 = false, false, false
				for _, otherPlayer in pairs(Players:GetPlayers()) do
					if otherPlayer ~= Player and otherPlayer.Character and otherPlayer.Character:FindFirstChild("HumanoidRootPart") then
						local distance = (otherPlayer.Character.HumanoidRootPart.Position - centerPos).Magnitude
						if isPlayerAuthorized(otherPlayer.Name) then
							if table.find(lobbyPlayers, otherPlayer.Name) then playerInZone1, playerInZone2 = true, true
							elseif otherPlayer.UserId < Player.UserId and distance <= 350 then playerInZone1, playerInZone2 = true, true end
						else
							if distance <= 75 then playerInZone1 = true end
							if distance <= 180 then playerInZone2 = true end
							if distance <= 350 then playerInZone3 = true end
						end
					end
				end

				local startsInText = ""
				local startsInObj = teleportModel:FindFirstChild("RaidPlayers") and teleportModel.RaidPlayers:FindFirstChild("SurfaceGui") and teleportModel.RaidPlayers.SurfaceGui:FindFirstChild("StartsIn")
				if startsInObj then startsInText = startsInObj.Text end
				local parsedSeconds = tonumber(string.match(startsInText, "%d+")) or 999
				local lobbyTimerActive = startsInText ~= "" and startsInText ~= "TELEPORTING PARTY"

				local function forceEvacuate()
					executeKingsRuinsLeave()
					pcall(function()
						local hrp = Player.Character.HumanoidRootPart; local offsetDir = (hrp.Position - centerPos).Unit
						if offsetDir.Magnitude ~= offsetDir.Magnitude or offsetDir.Magnitude == 0 then offsetDir = Vector3.new(1,0,0) end
						hrp.CFrame = CFrame.new(centerPos + (offsetDir * 70))
					end)
				end

				if amInLobby then
					if areOthersInLobby then updateUIRunsStatus("Strangers inside lobby | Evacuating") forceEvacuate(); task.wait(2) return end
					if playerInZone1 then if not lobbyTimerActive or parsedSeconds > 2 then updateUIRunsStatus("Player/Alt in CORE ZONE! | Evacuating"); forceEvacuate(); task.wait(2) return end
					elseif playerInZone2 then if not lobbyTimerActive or parsedSeconds > 15 then updateUIRunsStatus("Player/Alt approaching Mid | Evacuating"); forceEvacuate(); task.wait(2) return end end
					updateUIRunsStatus("Waiting for start (" .. startsInText .. ")")
				else
					if playerInZone2 then updateUIRunsStatus("Sector blocked by player/alt | Waiting hidden") else
						if startsInText == "TELEPORTING PARTY" then updateUIRunsStatus("Waiting for portal status to clear") else updateUIRunsStatus("Entering Lobby (Coast clear)") executeKingsRuinsEntrance() end
					end
				end
			end)
			task.wait(0.2)
		end
	end)
end
